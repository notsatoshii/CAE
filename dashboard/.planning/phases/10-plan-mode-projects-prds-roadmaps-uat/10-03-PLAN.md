```typescript
import type { Project } from "./cae-types";

export interface UatItem {
  /** sha1(phase + ':' + text).slice(0,8) */
  id: string;
  /** original bullet text */
  label: string;
  status: "pending" | "pass" | "fail";
  /** optional free-text note from the UAT reviewer */
  note?: string;
  /** ISO timestamp of last status change */
  ts?: string;
  /** true when a previously-known id no longer appears in the current ROADMAP */
  orphaned?: boolean;
}

export interface UatState {
  phase: number;
  items: UatItem[];
}

/**
 * Parse all '## Phase N:' sections; for each, extract 'Definition of done:' bullet list;
 * return Map<phaseNum, UatItem[]> with status='pending' on every item.
 * Case-insensitive match on 'Definition of done'. Phase numbers parsed as integers.
 */
export function parseSuccessCriteria(md: string): Map<number, UatItem[]>;

/**
 * Load or initialize <proj>/.planning/uat/phase<N>.json.
 * When file missing, derives fresh items from current ROADMAP.md via parseSuccessCriteria.
 * When file present, merges: known-id statuses preserved, missing-in-ROADMAP items flagged orphaned, new ROADMAP items appended as pending.
 */
export function loadUatState(proj: Project, phaseNum: number): Promise<UatState>;

/**
 * Update a single item's status + optional note, persist to disk.
 * Throws when id not found in loaded state.
 */
export function patchUatState(proj: Project, phaseNum: number, id: string, status: "pass" | "fail", note?: string): Promise<UatState>;
```

```typescript
import { spawn } from "child_process";
import { mkdir, readdir, writeFile } from "fs/promises";
import { join } from "path";
import type { Project } from "./cae-types";

const CAE_ROOT = "/home/cae/ctrl-alt-elite";
const CAE_ARCH_PERSONA = join(CAE_ROOT, ".claude/skills/cae-arch/persona.md"); // or SKILL.md — verify which one cae-planner's system-prompt actually uses by reading .claude/skills/cae-arch/
const PLAN_GEN_MODEL = "claude-opus-4-7";
const PLAN_GEN_TIMEOUT_LABEL = "10min"; // tmux session name hint; actual timeout enforced by caller

export interface PlanGenResult { /* see interfaces block */ }

function slugFromHeading(line: string): string {
  const m = /^##\s+Phase\s+\d+:?\s*(.*)$/i.exec(line.trim());
  const text = (m?.[1] ?? "").toLowerCase();
  const slug = text.replace(/[^a-z0-9]+/g, "-").replace(/^-+|-+$/g, "");
  return slug || "phase-one";
}

export function extractPhase1(md: string): string | null {
  const re = /^##\s+Phase\s+1\b.*$([\s\S]*?)(?=^##\s+Phase\s+\d+\b|\Z)/m;
  const m = re.exec(md);
  if (!m) return null;
  // m[0] starts with "## Phase 1..." and continues to next "## Phase" boundary.
  return m[0].trim();
}

export async function writeBuildplan(proj: Project, phase1Text: string): Promise<string> {
  const firstLine = phase1Text.split("\n", 1)[0] ?? "";
  const slug = slugFromHeading(firstLine);
  const dir = join(proj.path, ".planning", "phases", `01-${slug}`);
  await mkdir(dir, { recursive: true });
  const file = join(dir, "BUILDPLAN.md");
  await writeFile(file, phase1Text, "utf8");
  return file;
}

export async function runPlanGen(proj: Project): Promise<PlanGenResult> {
  // Locate the BUILDPLAN.md written by writeBuildplan.
  const phasesDir = join(proj.path, ".planning", "phases");
  const entries = await readdir(phasesDir).catch(() => [] as string[]);
  const phase01 = entries.find((e) => /^01-/.test(e));
  if (!phase01) {
    // No buildplan → fall back to stub immediately.
    const planPath = await stubPlan(proj);
    return { spawned: false, planPath };
  }
  const planDir = join(phasesDir, phase01);
  const planPath = join(planDir, "PLAN.md");
  const logFile = join(planDir, "plan-gen.log");
  const sid = `plan-gen-${phase01}-${Date.now().toString(36)}`;

  const prompt = `Draft PLAN.md in this directory from BUILDPLAN.md. Follow the GSD plan structure. Output to ${quote(planPath)}. Use opus-4-7 quality.`;
  const inner = `cd ${quote(planDir)} && claude --print --append-system-prompt-file ${quote(CAE_ARCH_PERSONA)} --model ${PLAN_GEN_MODEL} ${quote(prompt)} > ${quote(planPath)} 2> ${quote(logFile)}`;
  try {
    const child = spawn("tmux", ["new-session", "-d", "-s", sid, inner], { detached: true, stdio: "ignore" });
    child.unref();
    return { spawned: true, sid, planPath };
  } catch {
    const path = await stubPlan(proj);
    return { spawned: false, planPath: path };
  }
}

export async function stubPlan(proj: Project): Promise<string> {
  const phasesDir = join(proj.path, ".planning", "phases");
  const entries = await readdir(phasesDir).catch(() => [] as string[]);
  const phase01 = entries.find((e) => /^01-/.test(e)) ?? "01-phase-one";
  const planDir = join(phasesDir, phase01);
  await mkdir(planDir, { recursive: true });
  const planPath = join(planDir, "PLAN.md");
  const body =
    "# WAITING FOR PLANS\n\n" +
    "Auto-generation failed. Run `/gsd-plan-phase 01` interactively.\n";
  await writeFile(planPath, body, "utf8");
  return planPath;
}

function quote(s: string): string {
  return `'` + s.replace(/'/g, `'\\''`) + `'`;
}
```

```typescript
import { createHash } from "crypto";
import { mkdir, readFile, writeFile } from "fs/promises";
import { join } from "path";
import type { Project } from "./cae-types";

export interface UatItem { /* see interfaces block */ }
export interface UatState { /* see interfaces block */ }

function hashId(phase: number, text: string): string {
  return createHash("sha1").update(`${phase}:${text}`).digest("hex").slice(0, 8);
}

export function parseSuccessCriteria(md: string): Map<number, UatItem[]> {
  const out = new Map<number, UatItem[]>();
  const phaseRe = /^##\s+Phase\s+(\d+)\b[^\n]*\n([\s\S]*?)(?=^##\s+Phase\s+\d+\b|\Z)/gm;
  for (const m of md.matchAll(phaseRe)) {
    const phaseNum = parseInt(m[1], 10);
    const body = m[2];
    const dodMatch = /Definition of done:\s*\n([\s\S]*?)(?:\n\s*\n|$)/i.exec(body);
    if (!dodMatch) continue;
    const bullets = [...dodMatch[1].matchAll(/^\s*-\s+(.+)$/gm)].map((b) => b[1].trim()).filter(Boolean);
    const items: UatItem[] = bullets.map((label) => ({
      id: hashId(phaseNum, label),
      label,
      status: "pending",
    }));
    if (items.length) out.set(phaseNum, items);
  }
  return out;
}

async function readRoadmap(proj: Project): Promise<string> {
  const p = join(proj.path, ".planning", "ROADMAP.md");
  return await readFile(p, "utf8").catch(() => "");
}

function stateFile(proj: Project, phaseNum: number): string {
  return join(proj.path, ".planning", "uat", `phase${phaseNum}.json`);
}

export async function loadUatState(proj: Project, phaseNum: number): Promise<UatState> {
  const file = stateFile(proj, phaseNum);
  const roadmapMd = await readRoadmap(proj);
  const parsed = parseSuccessCriteria(roadmapMd).get(phaseNum) ?? [];

  let existing: UatState | null = null;
  try {
    existing = JSON.parse(await readFile(file, "utf8")) as UatState;
  } catch { /* new file */ }

  let items: UatItem[];
  if (!existing) {
    items = parsed;
  } else {
    // Merge: known ids keep their status + note + ts; new parsed ids appended; missing parsed ids flagged orphaned.
    const parsedIds = new Set(parsed.map((p) => p.id));
    const existingById = new Map(existing.items.map((it) => [it.id, it]));
    items = [
      ...parsed.map((p) => {
        const prior = existingById.get(p.id);
        return prior ? { ...p, status: prior.status, note: prior.note, ts: prior.ts } : p;
      }),
      ...existing.items
        .filter((it) => !parsedIds.has(it.id))
        .map((it) => ({ ...it, orphaned: true })),
    ];
  }

  const state: UatState = { phase: phaseNum, items };
  await mkdir(join(proj.path, ".planning", "uat"), { recursive: true });
  await writeFile(file, JSON.stringify(state, null, 2), "utf8");
  return state;
}

export async function patchUatState(
  proj: Project,
  phaseNum: number,
  id: string,
  status: "pass" | "fail",
  note?: string,
): Promise<UatState> {
  const state = await loadUatState(proj, phaseNum);
  const idx = state.items.findIndex((it) => it.id === id);
  if (idx < 0) throw new Error(`uat item ${id} not found in phase ${phaseNum}`);
  state.items[idx] = {
    ...state.items[idx],
    status,
    note,
    ts: new Date().toISOString(),
  };
  await writeFile(stateFile(proj, phaseNum), JSON.stringify(state, null, 2), "utf8");
  return state;
}
```
