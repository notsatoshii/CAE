```typescript
// dashboard/lib/cae-shift.ts

import type { Project } from "./cae-types";

/** Shift lifecycle phase string. Strings come straight from .shift/state.json::phase. */
export type ShiftPhase =
  | "idea" | "research" | "prd" | "roadmap"
  | "waiting_for_plans" | "executing" | "done";

/** Shape of .shift/state.json — subset we care about. Unknown fields preserved via `extra`. */
export interface ShiftState {
  schema_version: number;
  project_name: string;
  phase: ShiftPhase | string;
  updated: string;
  idea?: { what?: string; who?: string; type?: string };
  prd?: { path?: string; user_approved?: boolean };
  roadmap?: { path?: string; user_approved?: boolean };
  history?: Array<{ ts: string; action: string; outcome: string }>;
  [key: string]: unknown; // preserve unknown keys across read/write
}

export interface WizardAnswers {
  /** Shift qid `idea.what` — one-line problem statement. */
  what: string;
  /** Shift qid `idea.who` — target user. */
  who: string;
  /** Shift qid `idea.type_ok` — project type, usually "web". */
  typeOk: string;
}

/**
 * Match `slugOrPath` against the listProjects() whitelist.
 * - Absolute paths must equal a Project.path exactly.
 * - Short slugs match path.basename(Project.path).
 * - Returns null on any mismatch (never throws).
 * - Never shell-escapes, never file-system touches — pure resolution.
 */
export function resolveProject(slugOrPath: string): Promise<Project | null>;

/** Read and JSON.parse .shift/state.json. Returns null when file absent. */
export function readShiftState(projectPath: string): Promise<ShiftState | null>;

/**
 * Write /tmp/shift-answers-<uuid>.json keyed by Shift's own question ids.
 * Returns the absolute path of the written file (safe to pass as SHIFT_ANSWERS env).
 */
export function buildAnswersFile(answers: WizardAnswers): Promise<string>;

/**
 * Spawn tmux-detached `shift new <name>` under projectsHome with SHIFT_NONINTERACTIVE=1
 * + SHIFT_ANSWERS=<answersFile>. Returns the tmux session id (caller polls via API).
 * Does NOT wait for completion — this is a 30-600s operation.
 */
export function runShiftNew(name: string, answersFile: string): Promise<{ sid: string; projectPath: string; logFile: string }>;

/** Spawn tmux-detached `shift next` inside proj.path. Returns tmux sid + log path. */
export function runShiftNext(proj: Project): Promise<{ sid: string; logFile: string }>;

/**
 * Patch state.prd.user_approved=true, advance phase=roadmap, append history,
 * then spawn `shift next` to trigger ROADMAP drafting. Throws if state.phase != "prd".
 */
export function approvePrdGate(proj: Project): Promise<{ sid: string }>;

/**
 * Patch state.roadmap.user_approved=true, advance phase="waiting_for_plans",
 * append history. Does NOT spawn shift next — D-09 plan-gen handles phase advance.
 * Throws if state.phase != "roadmap".
 */
export function approveRoadmapGate(proj: Project): Promise<void>;
```

```typescript
function quote(s: string): string {
  return `'` + s.replace(/'/g, `'\\''`) + `'`;
```

```typescript
export async function resolveProject(slugOrPath: string): Promise<Project | null> {
  if (typeof slugOrPath !== "string" || slugOrPath.length === 0) return null;
  const projects = await listProjects();
  // Absolute path match
  if (slugOrPath.startsWith("/")) {
    const abs = resolve(slugOrPath);
    return projects.find((p) => p.path === abs) ?? null;
  }
  // Basename slug match (reject any slash entirely — no `../x`)
  if (slugOrPath.includes("/") || slugOrPath.includes("..")) return null;
  return projects.find((p) => basename(p.path) === slugOrPath) ?? null;
}

export async function readShiftState(projectPath: string): Promise<ShiftState | null> {
  const stateFile = join(projectPath, ".shift", "state.json");
  try {
    const raw = await readFile(stateFile, "utf8");
    return JSON.parse(raw) as ShiftState;
  } catch (err: unknown) {
    if ((err as NodeJS.ErrnoException).code === "ENOENT") return null;
    throw err; // malformed JSON or perms — propagate
  }
}
```

```typescript
const SHIFT_PROJECTS_HOME = process.env.SHIFT_PROJECTS_HOME ?? "/home/cae";
const SHIFT_BIN = "/home/shift/bin/shift";

export async function buildAnswersFile(a: WizardAnswers): Promise<string> {
  const file = join(tmpdir(), `shift-answers-${randomUUID()}.json`);
  const payload = {
    "idea.what": a.what,
    "idea.who": a.who,
    "idea.type_ok": a.typeOk,
  };
  await writeFile(file, JSON.stringify(payload, null, 2), { mode: 0o600, encoding: "utf8" });
  return file;
}

export async function runShiftNew(name: string, answersFile: string): Promise<{ sid: string; projectPath: string; logFile: string }> {
  if (!/^[a-zA-Z0-9_-]{1,64}$/.test(name)) {
    throw new Error("invalid project name");
  }
  const projectPath = join(SHIFT_PROJECTS_HOME, name);
  await mkdir(projectPath, { recursive: true });
  const sid = `plan-new-${name}-${Date.now().toString(36)}`;
  const logFile = join(projectPath, `.shift-bootstrap.log`);
  const inner = `cd ${quote(SHIFT_PROJECTS_HOME)} && SHIFT_NONINTERACTIVE=1 SHIFT_ANSWERS=${quote(answersFile)} ${SHIFT_BIN} new ${quote(name)} --dir ${quote(projectPath)} 2>&1 | tee ${quote(logFile)}`;
  const child = spawn("tmux", ["new-session", "-d", "-s", sid, inner], { detached: true, stdio: "ignore" });
  child.unref();
  return { sid, projectPath, logFile };
}

export async function runShiftNext(proj: Project): Promise<{ sid: string; logFile: string }> {
  const sid = `plan-next-${basename(proj.path)}-${Date.now().toString(36)}`;
  const logFile = join(proj.path, ".shift", `next-${Date.now()}.log`);
  const inner = `cd ${quote(proj.path)} && SHIFT_NONINTERACTIVE=1 ${SHIFT_BIN} next 2>&1 | tee ${quote(logFile)}`;
  const child = spawn("tmux", ["new-session", "-d", "-s", sid, inner], { detached: true, stdio: "ignore" });
  child.unref();
  return { sid, logFile };
}
```

```typescript
async function patchState(proj: Project, mutator: (s: ShiftState) => void): Promise<ShiftState> {
  const stateFile = join(proj.path, ".shift", "state.json");
  const raw = await readFile(stateFile, "utf8");
  const state = JSON.parse(raw) as ShiftState;
  const ts = new Date().toISOString().replace(/\.\d+Z$/, "Z");
  mutator(state);
  state.updated = ts;
  await writeFile(stateFile, JSON.stringify(state, null, 2), "utf8");
  return state;
}

function appendHistory(s: ShiftState, action: string, outcome: string): void {
  const ts = s.updated;
  if (!Array.isArray(s.history)) s.history = [];
  s.history.push({ ts, action, outcome });
}

export async function approvePrdGate(proj: Project): Promise<{ sid: string }> {
  const existing = await readShiftState(proj.path);
  if (!existing) throw new Error("no .shift/state.json");
  if (existing.phase !== "prd") {
    throw new Error(`state.phase must be 'prd' to approve PRD, got ${existing.phase}`);
  }
  await patchState(proj, (s) => {
    s.prd = { ...(s.prd ?? {}), user_approved: true };
    s.phase = "roadmap";
    appendHistory(s, "prd_approved", "dashboard");
  });
  const { sid } = await runShiftNext(proj);
  return { sid };
}

export async function approveRoadmapGate(proj: Project): Promise<void> {
  const existing = await readShiftState(proj.path);
  if (!existing) throw new Error("no .shift/state.json");
  if (existing.phase !== "roadmap") {
    throw new Error(`state.phase must be 'roadmap' to approve ROADMAP, got ${existing.phase}`);
  }
  await patchState(proj, (s) => {
    s.roadmap = { ...(s.roadmap ?? {}), user_approved: true };
    s.phase = "waiting_for_plans";
    appendHistory(s, "roadmap_approved", "dashboard");
  });
  // Intentionally does NOT call runShiftNext — plan-gen handles the phase.
}
```
