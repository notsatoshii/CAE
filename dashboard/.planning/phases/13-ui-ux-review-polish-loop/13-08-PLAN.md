     ```typescript
     import { createReadStream } from "node:fs";
     import { watch, stat } from "node:fs/promises";
     import { createInterface } from "node:readline";
     import { FSWatcher } from "node:fs";

     export type LogLine = { level: string; time: number; scope?: string; reqId?: string; msg?: string; [k: string]: unknown };
     export type LineHandler = (line: LogLine, raw: string) => void;

     export function filterLevel(min: "warn" | "error") {
       return (l: LogLine) => {
         if (min === "error") return l.level === "error" || l.level === "fatal";
         return l.level === "warn" || l.level === "error" || l.level === "fatal";
       };
     }

     export async function tailJsonl(
       file: string,
       opts: { onLine: LineHandler; onClose?: () => void; filter?: (l: LogLine) => boolean; historyLimit?: number; signal?: AbortSignal },
     ): Promise<() => void> {
       const filter = opts.filter ?? (() => true);
       const historyLimit = opts.historyLimit ?? 50;

       // emit last N lines matching filter
       try {
         const rl = createInterface({ input: createReadStream(file), crlfDelay: Infinity });
         const history: Array<{ line: LogLine; raw: string }> = [];
         for await (const raw of rl) {
           try {
             const parsed = JSON.parse(raw) as LogLine;
             if (filter(parsed)) history.push({ line: parsed, raw });
           } catch { /* skip malformed */ }
         }
         const recent = history.slice(-historyLimit);
         for (const h of recent) opts.onLine(h.line, h.raw);
       } catch {
         // file may not exist yet; ok, start tailing for new writes
       }

       // now tail: re-stat + re-open from last position on each change
       let lastSize = 0;
       try { lastSize = (await stat(file)).size; } catch {}

       let closed = false;
       const close = () => { closed = true; opts.onClose?.(); };
       if (opts.signal) opts.signal.addEventListener("abort", close);

       const pollTail = async () => {
         while (!closed) {
           try {
             const s = await stat(file);
             if (s.size > lastSize) {
               const chunk = createReadStream(file, { start: lastSize });
               const rl = createInterface({ input: chunk, crlfDelay: Infinity });
               for await (const raw of rl) {
                 if (closed) return;
                 try {
                   const parsed = JSON.parse(raw) as LogLine;
                   if (filter(parsed)) opts.onLine(parsed, raw);
                 } catch { /* skip */ }
               }
               lastSize = s.size;
             }
           } catch { /* transient errors ok */ }
           await new Promise(r => setTimeout(r, 500));
         }
       };
       pollTail();

       return close;
     }
     ```
     (If `fs.watch` is more reliable on the dev OS, use it — but stat+poll is simpler and Linux-safe for our logs.)

     ```typescript
     export type ClientLogEntry = { time: number; level: "debug"|"info"|"warn"|"error"; scope: string; msg: string; ctx?: Record<string, unknown> };
     const BUFFER: ClientLogEntry[] = [];
     const CAPACITY = 50;

     export function clientLog(level: ClientLogEntry["level"], scope: string, msg: string, ctx?: Record<string, unknown>) {
       const entry: ClientLogEntry = { time: Date.now(), level, scope, msg, ctx };
       BUFFER.push(entry);
       if (BUFFER.length > CAPACITY) BUFFER.shift();
       if (typeof window !== "undefined") {
         window.dispatchEvent(new CustomEvent("cae:log", { detail: entry }));
       }
     }

     export function getBuffer(): ClientLogEntry[] { return [...BUFFER]; }
     export function clearBuffer() { BUFFER.length = 0; }
     ```

     "use client";
     import { useEffect, useState } from "react";
     import { useDevMode } from "@/lib/hooks/use-dev-mode";
     import { getBuffer, ClientLogEntry } from "@/lib/client-log-bus";

     export function DebugBreadcrumbPanel() {
       const devMode = useDevMode();
       const [entries, setEntries] = useState<ClientLogEntry[]>(() => getBuffer());
       const [expanded, setExpanded] = useState(false);
       const [selected, setSelected] = useState<ClientLogEntry | null>(null);

       useEffect(() => {
         if (!devMode) return;
         const onLog = (ev: Event) => {
           const entry = (ev as CustomEvent).detail as ClientLogEntry;
           setEntries(b => [...b, entry].slice(-50));
         };
         window.addEventListener("cae:log", onLog);
         return () => window.removeEventListener("cae:log", onLog);
       }, [devMode]);

       if (!devMode) return null;
       return (
         <div className="fixed bottom-2 right-2 z-50 text-[10px] font-mono">
           <button onClick={() => setExpanded(v => !v)} className="rounded border border-[color:var(--border)] bg-[color:var(--bg-elev)] px-2 py-1">
             breadcrumbs ({entries.length}){expanded ? " ▼" : " ▲"}
           </button>
           {expanded && (
             <div className="mt-1 max-h-64 w-80 overflow-auto rounded border border-[color:var(--border)] bg-[color:var(--bg-elev)] p-2">
               {entries.length === 0 ? <div className="text-[color:var(--text-dim)]">no events yet</div> :
                 entries.slice().reverse().map((e, i) => (
                   <div key={i} onClick={() => setSelected(e)} className="cursor-pointer hover:bg-[color:var(--bg)]">

                     {" "}<span>{e.msg}</span>
                   </div>
                 ))
               }
               {selected && (
                 <pre className="mt-2 overflow-auto rounded bg-[color:var(--bg)] p-1 text-[10px]">{JSON.stringify(selected, null, 2)}</pre>
               )}
             </div>
           )}
         </div>
       );
     }
