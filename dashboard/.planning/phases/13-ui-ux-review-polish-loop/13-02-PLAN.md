  Then capture a console baseline per protected route for Wave 3's logging audit. Don't re-use capture.sh — write a tiny one-off:
  ```bash
  python3 - <<'PY' > ../audit/working/console-baseline.tsv
  import json, pathlib
  from playwright.sync_api import sync_playwright
  ROUTES = json.loads(pathlib.Path("routes.json").read_text())
  with sync_playwright() as pw:
      browser = pw.chromium.launch(headless=True)
      print("slug\tconsole_error_count\tconsole_warn_count\tpageerror_count\tfirst_error_sample")
      for route in ROUTES["routes"]:
          if not route.get("auth"): continue
          ctx = browser.new_context(
              viewport={"width":1280,"height":800}, device_scale_factor=1,
              storage_state="storage-state.json",
          )
          ctx.add_init_script("localStorage.setItem('explainMode','true'); localStorage.setItem('devMode','false')")
          page = ctx.new_page()
          errs = []; warns = []; pe = []
          page.on("console", lambda m, e=errs, w=warns: e.append(m.text) if m.type == "error" else (w.append(m.text) if m.type == "warning" else None))
          page.on("pageerror", lambda exc, p=pe: p.append(str(exc)))
          try:
              page.goto(ROUTES["base_url"] + route["path"], wait_until="domcontentloaded", timeout=30000)
              if route.get("wait_selector"):
                  try: page.wait_for_selector(route["wait_selector"], timeout=10000)
                  except: pass
              page.wait_for_timeout(int(route.get("wait_ms", 800)) + 2000)  # extra settle to let polling fire
          except Exception as e:
              errs.append(f"navigation failed: {e}")
          sample = (errs + pe + [""])[0].replace("\n"," ").replace("\t"," ")[:200]
          print(f"{route['slug']}\t{len(errs)}\t{len(warns)}\t{len(pe)}\t{sample}")
          ctx.close()
      browser.close()
  PY
  ```
  Result: `audit/working/console-baseline.tsv` — 1 row per protected route. Plan 13-05 (logging) reads this to decide what's noise vs signal.
