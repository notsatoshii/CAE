**encodeSSE contract update (lib/sse.ts):**
```typescript
/** encodeSSE — id MUST be non-empty iff the frame corresponds to a persisted message id
 *  that the client can safely promote to lastSeenMsgId. Ephemeral frames (deltas, ticks)
 *  MUST pass id="".
 */
export function encodeSSE(id: string, event: string, data: unknown): string {
  const parts = [];
  if (id) parts.push(`id: ${id}`);
  parts.push(`event: ${event}`);
  parts.push(`data: ${JSON.stringify(data)}`);
  return parts.join("\n") + "\n\n";
}
```
