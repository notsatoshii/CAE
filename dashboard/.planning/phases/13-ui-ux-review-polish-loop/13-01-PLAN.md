  Write `.planning/phases/13-ui-ux-review-polish-loop/scripts/capture.sh`:
  ```bash
  #!/usr/bin/env bash
  set -euo pipefail
  PHASE_ARG="${1:-}"; shift || true
  if [[ "$PHASE_ARG" != "before" && "$PHASE_ARG" != "after" ]]; then
    echo "usage: capture.sh before|after [--route slug] [--viewport mobile|laptop|wide] [--mode founder|dev]" >&2
    exit 2
  fi
  ROUTE_FILTER=""; VIEWPORT_FILTER=""; MODE_FILTER=""
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --route)    ROUTE_FILTER="$2"; shift 2 ;;
      --viewport) VIEWPORT_FILTER="$2"; shift 2 ;;
      --mode)     MODE_FILTER="$2"; shift 2 ;;
      *) echo "unknown arg: $1" >&2; exit 2 ;;
    esac
  done
  cd "$(dirname "$0")"
  export PLAYWRIGHT_BROWSERS_PATH="/usr/local/share/playwright-browsers"
  export PHASE="$PHASE_ARG" ROUTE_FILTER VIEWPORT_FILTER MODE_FILTER
  python3 - <<'PY'
  import json, os, sys, pathlib
  from playwright.sync_api import sync_playwright

  ROUTES = json.loads(pathlib.Path("routes.json").read_text())
  PHASE = os.environ["PHASE"]
  ROUTE_FILTER = os.environ.get("ROUTE_FILTER","")
  VIEWPORT_FILTER = os.environ.get("VIEWPORT_FILTER","")
  MODE_FILTER = os.environ.get("MODE_FILTER","")
  SHOTS = pathlib.Path(f"../shots/{PHASE}")
  SHOTS.mkdir(parents=True, exist_ok=True)
  MANIFEST = SHOTS / "MANIFEST.tsv"
  DETERMINISTIC_CSS = "* { animation: none !important; transition: none !important; caret-color: transparent !important; } *::before, *::after { animation: none !important; transition: none !important; }"
  fails = 0; shots = 0
  with sync_playwright() as pw:
      browser = pw.chromium.launch(headless=True)
      with MANIFEST.open("w") as mf:
          mf.write("slug\tviewport\tmode\turl\tpng_path\tbytes\n")
          for mode_name, matrix_viewports in ROUTES["mode_viewport_matrix"].items():
              if MODE_FILTER and mode_name != MODE_FILTER: continue
              mode_cfg = ROUTES["modes"][mode_name]
              for vp_name in matrix_viewports:
                  if VIEWPORT_FILTER and vp_name != VIEWPORT_FILTER: continue
                  w,h = map(int, ROUTES["viewports"][vp_name].split("x"))
                  out_dir = SHOTS / f"{vp_name}-{mode_name}"
                  out_dir.mkdir(parents=True, exist_ok=True)
                  for route in ROUTES["routes"] + ROUTES["drawer_states"]:
                      if ROUTE_FILTER and route["slug"] != ROUTE_FILTER: continue
                      if route in ROUTES["drawer_states"] and (vp_name != "laptop" or mode_name != "founder"): continue
                      ctx_kwargs = {"viewport": {"width":w,"height":h}, "device_scale_factor": 1}  # V2 §7: 1x only for Opus 4.7 3.75MP cap
                      if route.get("auth"):
                          ctx_kwargs["storage_state"] = "storage-state.json"
                      ctx = browser.new_context(**ctx_kwargs)
                      ctx.add_init_script(f"""
                          localStorage.setItem('explainMode','{mode_cfg['explainMode']}');
                          localStorage.setItem('devMode','{mode_cfg['devMode']}');
                      """)
                      page = ctx.new_page()
                      url = ROUTES["base_url"] + route["path"]
                      try:
                          page.goto(url, wait_until="domcontentloaded", timeout=30000)
                          if route.get("wait_selector"):
                              page.wait_for_selector(route["wait_selector"], timeout=15000)
                          page.wait_for_timeout(int(route.get("wait_ms", 800)))
                          page.add_style_tag(content=DETERMINISTIC_CSS)
                          page.wait_for_timeout(200)
                          png_path = out_dir / f"{route['slug']}.png"
                          page.screenshot(path=str(png_path), full_page=False)
                          size = png_path.stat().st_size
                          mf.write(f"{route['slug']}\t{vp_name}\t{mode_name}\t{url}\t{png_path}\t{size}\n")
                          shots += 1
                          print(f"[capture] {vp_name}/{mode_name}/{route['slug']} -> {size}B")
                      except Exception as e:
                          print(f"[capture:FAIL] {vp_name}/{mode_name}/{route['slug']}: {e}", file=sys.stderr)
                          fails += 1
                      finally:
                          ctx.close()
      browser.close()
  print(f"[capture] phase={PHASE} shots={shots} fails={fails}")
  sys.exit(1 if fails > 0 else 0)
  PY
  ```

  #!/usr/bin/env python3
  """Phase 13 per-route click walker. Drives every button + role=button + link, records console errors.

  Usage:
      cd scripts && python3 clickwalk.py --slug build-home
  Exit 0 = no console.error fired, 1 = errors detected.
  """
  import argparse, json, pathlib, sys
  from playwright.sync_api import sync_playwright

  SCRIPTS = pathlib.Path(__file__).parent
  ROUTES = json.loads((SCRIPTS / "routes.json").read_text())
  AUDIT_WORKING = SCRIPTS.parent / "audit" / "working"
  AUDIT_WORKING.mkdir(parents=True, exist_ok=True)

  parser = argparse.ArgumentParser()
  parser.add_argument("--slug", required=True)
  parser.add_argument("--mode", default="founder")
  parser.add_argument("--viewport", default="laptop")
  args = parser.parse_args()

  route = next((r for r in ROUTES["routes"] + ROUTES["drawer_states"] if r["slug"] == args.slug), None)
  if route is None:
      print(f"[clickwalk] slug '{args.slug}' not in routes.json", file=sys.stderr)
      sys.exit(2)

  w, h = map(int, ROUTES["viewports"][args.viewport].split("x"))
  mode_cfg = ROUTES["modes"][args.mode]
  url = ROUTES["base_url"] + route["path"]

  rows: list[str] = ["| # | Element | Aria/Text | Clicked | Console Errors | Verdict |", "|---|---|---|---|---|---|"]
  err_count = 0

  with sync_playwright() as pw:
      browser = pw.chromium.launch(headless=True)
      ctx_kwargs = {"viewport": {"width": w, "height": h}, "device_scale_factor": 1}
      if route.get("auth"):
          ctx_kwargs["storage_state"] = str(SCRIPTS / "storage-state.json")
      ctx = browser.new_context(**ctx_kwargs)
      ctx.add_init_script(f"""
          localStorage.setItem('explainMode','{mode_cfg['explainMode']}');
          localStorage.setItem('devMode','{mode_cfg['devMode']}');
      """)
      page = ctx.new_page()
      console_errors: list[str] = []
      page.on("console", lambda m: console_errors.append(m.text) if m.type == "error" else None)
      page.on("pageerror", lambda exc: console_errors.append(f"pageerror: {exc}"))
      page.goto(url, wait_until="domcontentloaded", timeout=30000)
      if route.get("wait_selector"):
          try: page.wait_for_selector(route["wait_selector"], timeout=10000)
          except: pass
      page.wait_for_timeout(int(route.get("wait_ms", 800)))

      elements = page.query_selector_all("button, [role='button'], a[href]")
      for i, el in enumerate(elements):
          tag = el.evaluate("e => e.tagName.toLowerCase()")
          aria = (el.get_attribute("aria-label") or (el.text_content() or "?")).strip()[:60]
          before = len(console_errors)
          clicked = "no"
          try:
              el.click(timeout=2000, force=False)
              clicked = "yes"
              page.wait_for_timeout(500)
          except Exception as e:
              clicked = f"FAIL: {str(e)[:40]}"
          new_errs = console_errors[before:]
          if new_errs: err_count += len(new_errs)
          verdict = "✅" if not new_errs and clicked == "yes" else ("⚠️" if not new_errs else "❌")
          rows.append(f"| {i+1} | `{tag}` | {aria} | {clicked} | {'; '.join(new_errs) if new_errs else '—'} | {verdict} |")
          try: page.keyboard.press("Escape")
          except: pass
          try: page.goto(url, wait_until="domcontentloaded", timeout=15000); page.wait_for_timeout(400)
          except: break
      ctx.close(); browser.close()

  out = AUDIT_WORKING / f"CLICKWALK-{args.slug}.md"
  out.write_text(
      f"# Clickwalk: {args.slug}\n\n"
      f"**Route:** {url}\n"
      f"**Viewport:** {args.viewport}\n"
      f"**Mode:** {args.mode}\n"
      f"**Elements:** {len(elements)}\n"
      f"**Console errors:** {err_count}\n\n"
      + "\n".join(rows) + "\n"
  )
  print(f"[clickwalk] {args.slug}: {len(elements)} elements, {err_count} console errors → {out}")
  sys.exit(0 if err_count == 0 else 1)
