     ```python
     import json, pathlib, sys
     from playwright.sync_api import sync_playwright
     ROUTES = json.loads(pathlib.Path("scripts/routes.json").read_text())
     AXE_CDN = "https://cdnjs.cloudflare.com/ajax/libs/axe-core/4.10.2/axe.min.js"
     results = []
     with sync_playwright() as pw:
       browser = pw.chromium.launch(headless=True)
       for route in ROUTES["routes"]:
         if not route.get("auth"): continue
         ctx = browser.new_context(viewport={"width":1280,"height":800}, storage_state="scripts/storage-state.json", device_scale_factor=1)
         ctx.add_init_script("localStorage.setItem('explainMode','true');localStorage.setItem('devMode','false')")
         page = ctx.new_page()
         page.goto(ROUTES["base_url"] + route["path"], wait_until="domcontentloaded", timeout=30000)
         page.wait_for_timeout(int(route.get("wait_ms", 800)))
         page.add_script_tag(url=AXE_CDN)
         res = page.evaluate("axe.run({runOnly:{type:'tag',values:['wcag2a','wcag2aa','wcag22aa']}})")
         results.append({"slug": route["slug"], "violations": res["violations"]})
         ctx.close()
       browser.close()
     print(json.dumps(results, indent=2))
     ```
