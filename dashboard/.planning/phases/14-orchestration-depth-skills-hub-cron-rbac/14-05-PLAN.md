       ```typescript
       import type { CatalogSkill, TrustScore, TrustFactor } from "./cae-types";
       import type { SkillFrontmatter } from "./cae-skills-parse";

       const TRUSTED_OWNERS = new Set(["anthropics", "vercel-labs", "anthropic", "anthropic-labs", "diiant"]);
       const RISKY_TOOL_RES = [/^Bash\(rm\b/, /^Bash\(curl\b/, /^Bash\(wget\b/, /^Bash\(\*\)$/, /^Bash\(sudo\b/];

       export type TrustInput = {
         skill: Pick<CatalogSkill, "owner" | "name">;
         frontmatter: SkillFrontmatter;
         secretsCount: number;
         docExampleCount?: number;  // excluded from "real findings"
         overridden?: boolean;
         updatedWithinDays?: number;
       };

       export function computeTrustScore(input: TrustInput): TrustScore {
         if (input.overridden) {
           return {
             total: 100,
             overridden: true,
             factors: [{ id: "admin_override", passed: true, weight: 1, reason: "Admin marked this skill as trusted" }],
           };
         }

         const factors: TrustFactor[] = [];

         // Factor 1: trusted owner
         const ownerTrusted = TRUSTED_OWNERS.has(input.skill.owner.toLowerCase());
         factors.push({
           id: "trusted_owner",
           passed: ownerTrusted,
           weight: 0.30,
           reason: ownerTrusted
             ? `${input.skill.owner} is on the trusted owners list`
             : `${input.skill.owner} is not on the trusted owners list — review code before installing`,
         });

         // Factor 2: allowed-tools declared
         const hasAllowed = (input.frontmatter.allowedTools?.length ?? 0) > 0;
         factors.push({
           id: "allowed_tools_declared",
           passed: hasAllowed,
           weight: 0.20,
           reason: hasAllowed
             ? "Skill declares which tools it can use"
             : "Skill does not declare allowed-tools — could run anything",
         });

         // Factor 3: no risky tools
         const risky = (input.frontmatter.allowedTools ?? []).filter(t => RISKY_TOOL_RES.some(re => re.test(t)));
         factors.push({
           id: "no_risky_tools",
           passed: risky.length === 0,
           weight: 0.20,
           reason: risky.length === 0
             ? "No risky tool patterns (rm, curl, wildcard Bash)"
             : `Uses risky tool patterns: ${risky.join(", ")}`,
         });

         // Factor 4: no secrets (excludes doc-examples)
         const realSecrets = Math.max(0, input.secretsCount - (input.docExampleCount ?? 0));
         factors.push({
           id: "no_secrets",
           passed: realSecrets === 0,
           weight: 0.20,
           reason: realSecrets === 0 ? "No secrets detected by scan" : `${realSecrets} secret-like strings detected`,
         });

         // Factor 5: recently updated
         const fresh = (input.updatedWithinDays ?? 999) <= 90;
         factors.push({
           id: "recently_updated",
           passed: fresh,
           weight: 0.10,
           reason: fresh ? "Updated within last 90 days" : "Not updated in 90+ days — may be stale",
         });

         const total = Math.round(factors.reduce((acc, f) => acc + (f.passed ? f.weight * 100 : 0), 0));
         return { total, factors };
       }
       ```

       ```typescript
       import { spawn } from "node:child_process";
       import { readFile } from "node:fs/promises";
       import path from "node:path";

       export type SecretFinding = { ruleId: string; file: string; startLine: number; matchRedacted: string; isDocExample: boolean };
       export type ScanResult = { available: boolean; findings: SecretFinding[]; error?: string; scannedAt: string };

       const DOC_EXAMPLES = [
         /your-api-key-here/i, /REDACTED/, /sk-proj-example/i, /xxxxxx+/, /example-key/i,
         /<your-[a-z-]+>/i, /{{[a-z_]+}}/i,
       ];

       export async function scanSkill(dir: string, spawnImpl = spawn): Promise<ScanResult> {
         const tmpReport = `/tmp/gitleaks-${Date.now()}-${Math.random().toString(36).slice(2)}.json`;
         return new Promise((resolve) => {
           const proc = spawnImpl("gitleaks", [
             "detect",
             "--no-git",
             "--source", dir,
             "--redact",
             "--report-format", "json",
             "--report-path", tmpReport,
             "--config", path.join(process.cwd(), "lib/gitleaks-allowlist.toml"),
           ]);
           let err = "";
           proc.stderr?.on("data", (d) => { err += d.toString(); });
           proc.on("error", (e) => resolve({ available: false, findings: [], error: (e as any).code === "ENOENT" ? "gitleaks not installed" : e.message, scannedAt: new Date().toISOString() }));
           proc.on("close", async (code) => {
             // gitleaks returns non-zero when findings present; 0 = clean. Only treat negative/spawn errors as failure.
             try {
               const raw = await readFile(tmpReport, "utf8").catch(() => "[]");
               const list = JSON.parse(raw || "[]") as any[];
               const findings: SecretFinding[] = list.map(r => ({
                 ruleId: r.RuleID ?? "unknown",
                 file: r.File ?? "",
                 startLine: r.StartLine ?? 0,
                 matchRedacted: r.Match ?? "",
                 isDocExample: DOC_EXAMPLES.some(re => re.test(r.Match ?? "")),
               }));
               resolve({ available: true, findings, scannedAt: new Date().toISOString() });
             } catch (e) {
               resolve({ available: true, findings: [], error: String((e as Error).message), scannedAt: new Date().toISOString() });
             }
           });
         });
       }

       export async function appendScan(name: string, report: ScanResult): Promise<void> { /* append-only JSONL to .cae/metrics/skill-scans.jsonl */ }
       ```

       ```typescript
       import { tailJsonl, listProjects } from "./cae-state";
       import path from "node:path";
       import type { AuditEntry } from "./cae-types";

       export type AuditFilter = { from?: string; to?: string; tool?: string; task?: string; limit?: number; offset?: number };

       export async function readAuditLog(filter: AuditFilter = {}): Promise<{ entries: AuditEntry[]; total: number }> {
         const projects = await listProjects();
         let all: AuditEntry[] = [];
         for (const p of projects) {
           try {
             const rows = await tailJsonl(path.join(p.path, ".cae/metrics/tool-calls.jsonl"), 10_000);
             for (const row of rows) {
               if (!isAuditEntry(row)) continue;
               all.push(row);
             }
           } catch { /* skip missing files */ }
         }
         // Apply filters
         if (filter.from) all = all.filter(e => e.ts >= filter.from!);
         if (filter.to) all = all.filter(e => e.ts <= filter.to!);
         if (filter.tool) all = all.filter(e => e.tool === filter.tool);
         if (filter.task) all = all.filter(e => e.task === filter.task);
         all.sort((a, b) => b.ts.localeCompare(a.ts));
         const total = all.length;
         const entries = all.slice(filter.offset ?? 0, (filter.offset ?? 0) + (filter.limit ?? 200));
         return { entries, total };
       }

       function isAuditEntry(v: unknown): v is AuditEntry { /* typeguard for ts/task/tool/cwd strings */ return !!v && typeof v === "object" && typeof (v as any).ts === "string" && typeof (v as any).tool === "string"; }
       ```

    5. lib/cae-trust-overrides.ts:

       ```typescript
       import { readFile, writeFile, chmod } from "node:fs/promises";
       import path from "node:path";

       const FILE = path.join(process.env.CAE_ROOT ?? "/home/cae/ctrl-alt-elite", ".cae/trust-overrides.json");

       export async function readOverrides(): Promise<Set<string>> {
         try { const raw = await readFile(FILE, "utf8"); return new Set(JSON.parse(raw) as string[]); }
         catch { return new Set(); }
       }

       export async function writeOverride(key: string, trusted: boolean): Promise<void> {
         const current = await readOverrides();
         if (trusted) current.add(key); else current.delete(key);
         const tmp = FILE + ".tmp";
         await writeFile(tmp, JSON.stringify([...current], null, 2));
         await chmod(tmp, 0o600);
         const { rename } = await import("node:fs/promises");
         await rename(tmp, FILE);
       }

       export function overrideKey(owner: string, name: string): string { return `${owner}/${name}`.toLowerCase(); }
       ```
