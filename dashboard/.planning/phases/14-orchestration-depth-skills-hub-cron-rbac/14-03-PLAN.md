       ```typescript
       type RuleHandler = (m: RegExpMatchArray) => string;
       const to24h = (h: string, meridiem?: string) => {
         let n = parseInt(h, 10);
         if (meridiem?.toLowerCase() === "pm" && n < 12) n += 12;
         if (meridiem?.toLowerCase() === "am" && n === 12) n = 0;
         return n;
       };
       const RULES: Array<[RegExp, RuleHandler | string]> = [
         [/^every (morning|day) at (\d+)(?::(\d+))?\s*(am|pm)?$/i, (m) => `${m[3] ?? 0} ${to24h(m[2], m[4] ?? "am")} * * *`],
         [/^every weekday at (\d+)(?::(\d+))?\s*(am|pm)?$/i, (m) => `${m[2] ?? 0} ${to24h(m[1], m[3] ?? "am")} * * 1-5`],
         [/^every (monday|tuesday|wednesday|thursday|friday|saturday|sunday) at (\d+)(?::(\d+))?\s*(am|pm)?$/i, (m) => { const dow = ["sunday","monday","tuesday","wednesday","thursday","friday","saturday"].indexOf(m[1].toLowerCase()); return `${m[3] ?? 0} ${to24h(m[2], m[4] ?? "am")} * * ${dow}`; }],
         [/^every (\d+) minutes?$/i, (m) => `*/${m[1]} * * * *`],
         [/^every hour$/i, "0 * * * *"],
         [/^every (\d+) hours?$/i, (m) => `0 */${m[1]} * * *`],
         [/^at midnight$/i, "0 0 * * *"],
         [/^at noon$/i, "0 12 * * *"],
         [/^every weekend at (\d+)(?::(\d+))?\s*(am|pm)?$/i, (m) => `${m[2] ?? 0} ${to24h(m[1], m[3] ?? "am")} * * 0,6`],
         // ... target ~20 rules covering ~80% of founder requests
       ];

       export type ParseResult = { cron: string; source: "rule" | "llm"; confidence: "high" | "medium" };
       export async function parseSchedule(nl: string, llmImpl = defaultLlm): Promise<ParseResult> {
         const trimmed = nl.trim();
         for (const [re, gen] of RULES) {
           const m = trimmed.match(re);
           if (m) {
             const cron = typeof gen === "function" ? gen(m) : gen;
             return { cron, source: "rule", confidence: "high" };
           }
         }
         const llm = await llmImpl(nl);
         const { default: CronParser } = await import("cron-parser");
         try {
           CronParser.parseExpression(llm.cron);
           return { cron: llm.cron, source: "llm", confidence: "medium" };
         } catch {
           throw new Error(`Could not parse: "${nl}"`);
         }
       }
       ```

       ```typescript
       import cronstrue from "cronstrue";
       import CronParser from "cron-parser";
       export function describeCron(cron: string, tz: string): { english: string; nextRun: Date | null } {
         let english: string;
         try { english = cronstrue.toString(cron); } catch { english = cron; }
         let nextRun: Date | null = null;
         try {
           const iter = CronParser.parseExpression(cron, { tz });
           nextRun = iter.next().toDate();
         } catch { /* swallow — invalid cron */ }
         return { english, nextRun };
       }
       ```

       ```typescript
       export const runtime = "nodejs";
       const bucket = new Map<string, { count: number; resetAt: number }>();
       function rateLimit(key: string, limit = 10, windowMs = 60_000): boolean {
         const now = Date.now();
         const b = bucket.get(key);
         if (!b || b.resetAt < now) { bucket.set(key, { count: 1, resetAt: now + windowMs }); return true; }
         if (b.count >= limit) return false;
         b.count++; return true;
       }
       export async function POST(req: NextRequest) {
         const ip = req.headers.get("x-forwarded-for") ?? "local";
         if (!rateLimit(ip)) return NextResponse.json({error:"rate limited"},{status:429});
         const { nl, timezone } = await req.json();
         if (typeof nl !== "string" || !nl.trim()) return NextResponse.json({error:"nl required"},{status:400});
         try {
           const parsed = await parseSchedule(nl);
           const { english, nextRun } = describeCron(parsed.cron, timezone ?? "UTC");
           return NextResponse.json({ ...parsed, english, nextRun: nextRun?.toISOString() ?? null });
         } catch (e) {
           return NextResponse.json({ error: String((e as Error).message) }, { status: 422 });
         }
       }
       ```

       ```bash
       #!/usr/bin/env bash
       set -euo pipefail
       CAE_ROOT="${CAE_ROOT:-/home/cae/ctrl-alt-elite}"
       TASKS_FILE="$CAE_ROOT/scheduled_tasks.json"
       LOG="$CAE_ROOT/.cae/metrics/scheduler.jsonl"
       LOCK_DIR="${LOCK_DIR:-/tmp}"
       mkdir -p "$(dirname "$LOG")"
       [[ -f "$TASKS_FILE" ]] || { echo "[]" > "$TASKS_FILE"; }

       NOW=$(date +%s)

       # Iterate each task
       jq -c '.[]' "$TASKS_FILE" | while IFS= read -r task; do
         id=$(echo "$task" | jq -r '.id')
         enabled=$(echo "$task" | jq -r '.enabled')
         cron=$(echo "$task" | jq -r '.cron')
         tz=$(echo "$task" | jq -r '.timezone // "UTC"')
         last=$(echo "$task" | jq -r '.lastRun // 0')
         buildplan=$(echo "$task" | jq -r '.buildplan')

         [[ "$enabled" == "true" ]] || continue

         # Compute next run after lastRun via node + cron-parser
         next=$(node -e "
           const c=require('cron-parser');
           const iter=c.parseExpression('$cron',{currentDate:new Date($last*1000),tz:'$tz'});
           console.log(Math.floor(iter.next().getTime()/1000));
         " 2>/dev/null || echo 0)

         if [[ "$NOW" -ge "$next" && "$next" -gt 0 ]]; then
           LOCK="$LOCK_DIR/cae-scheduler-$id.lock"
           # Non-blocking flock; skip if another watcher holds it
           if ! ( flock -n 9 ; true ) 9>"$LOCK"; then
             echo "{\"ts\":$NOW,\"event\":\"skip_locked\",\"id\":\"$id\"}" >> "$LOG"
             continue
           fi

           # Acquire lock for dispatch — run task in subshell holding the lock
           (
             flock -n 9 || exit 0
             echo "{\"ts\":$NOW,\"event\":\"dispatch\",\"id\":\"$id\"}" >> "$LOG"
             # Update lastRun BEFORE spawn (pitfall 7)
             tmp=$(mktemp)
             jq --arg id "$id" --argjson ts "$NOW" 'map(if .id==$id then .lastRun=$ts else . end)' "$TASKS_FILE" > "$tmp"
       ```
