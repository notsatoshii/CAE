       import { spawn } from "node:child_process";
       const REPO_RE = /^[A-Za-z0-9_.-]+\/[A-Za-z0-9_.-]+$/;
       const URL_OK = /^https:\/\/github\.com\/[A-Za-z0-9_.-]+\/[A-Za-z0-9_.-]+(\/tree\/[A-Za-z0-9_./-]+)?$/;
       export type InstallEvent = { type: "line" | "err" | "done"; data: string };
       export async function* installSkill(repo: string, spawnImpl = spawn): AsyncIterable<InstallEvent> {
         if (!REPO_RE.test(repo) && !URL_OK.test(repo)) throw new Error(`invalid repo: ${repo}`);
         const proc = spawnImpl("npx", ["-y", "skills", "add", repo], { env: { ...process.env, SKILLS_TELEMETRY_DISABLED: "1" } });
         // race stdout + stderr via two AsyncIterators + shared queue pattern (or merge via events)
         // simplest: use readline + two async-iterables merged with a queue

       }

    3. Create app/api/skills/install/route.ts — SSE streaming:
       ```typescript
       export const runtime = "nodejs";
       export async function POST(req: NextRequest) {
         const { repo } = await req.json();
         if (typeof repo !== "string") return NextResponse.json({error:"repo required"},{status:400});
         let stream;
         try { stream = installSkill(repo); } catch (e) { return NextResponse.json({error:String((e as Error).message)},{status:400}); }
         const enc = new TextEncoder();
         const body = new ReadableStream({
           async start(controller) {
             try {
               for await (const ev of stream) {
                 controller.enqueue(enc.encode(`event: ${ev.type}\ndata: ${JSON.stringify(ev.data)}\n\n`));
               }
             } finally { controller.close(); }
           }
         });
         return new Response(body, { headers: { "Content-Type": "text/event-stream", "Cache-Control": "no-cache, no-transform", "Connection": "keep-alive" } });
       }
       ```
