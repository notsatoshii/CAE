       ```ts

       // Outside useEffect:
       const kb = keybindingById("explain.toggle");
       // Inside useEffect:
       if (!kb) return;                         // defensive: registry hole = no-op, don't crash
       // kb.keys = ["Ctrl","E"] — derive expected-key from the last chip, modifiers from earlier chips.
       const expectedKey = kb.keys[kb.keys.length - 1].toLowerCase();    // "e"
       const wantCtrl  = kb.keys.includes("Ctrl");
       const wantMeta  = kb.keys.includes("⌘");
       const wantShift = kb.keys.includes("Shift");
       const wantAlt   = kb.keys.includes("Alt");
       if (e.key.toLowerCase() !== expectedKey) return;
       if (Boolean(e.ctrlKey)  !== wantCtrl)  return;
       if (Boolean(e.metaKey)  !== wantMeta)  return;
       if (Boolean(e.shiftKey) !== wantShift) return;
       if (Boolean(e.altKey)   !== wantAlt)   return;
       ```
       Extract the modifier-match logic into a tiny helper in `lib/keybindings.ts`:
       ```ts
       export function matchesKeydown(kb: Keybinding, e: KeyboardEvent): boolean {
         const expectedKey = kb.keys[kb.keys.length - 1].toLowerCase();
         if (e.key.toLowerCase() !== expectedKey) return false;
         if (Boolean(e.ctrlKey)  !== kb.keys.includes("Ctrl"))  return false;
         if (Boolean(e.metaKey)  !== kb.keys.includes("⌘"))    return false;
         if (Boolean(e.shiftKey) !== kb.keys.includes("Shift")) return false;
         if (Boolean(e.altKey)   !== kb.keys.includes("Alt"))   return false;
         return true;
       }
       ```
       Add unit tests for `matchesKeydown` in `lib/keybindings.test.ts` (extend Wave-0 file; do NOT create a duplicate file). Include at least: Ctrl+E match; ⌘Shift+D match; rejection when modifier bitmap differs; case-insensitive letter key; `?` single-key match; Esc match.

       ```ts
       import { keybindingById, matchesKeydown } from "@/lib/keybindings";
       // …

         if (!kb) return;
         function onKeyDown(e: KeyboardEvent) {
           if (isEditableTarget(e.target)) return;
           if (!matchesKeydown(kb, e)) return;
           e.preventDefault();
           setExplainState((prev) => !prev);
         }
         window.addEventListener("keydown", onKeyDown);
         return () => window.removeEventListener("keydown", onKeyDown);
       }, []);
       ```
