```typescript
export type StepAgent   = { agent: "forge" | "sentinel" | "scout" | "scribe" | "phantom" | "aegis" | "arch" | "herald" | "nexus"; task: string; timeout?: string }
export type StepGate    = { gate: "approval" | "auto"; notify?: "telegram" | "email" }
export type StepAction  = { action: "push" | "abort" | "branch" }
export type WorkflowStep = StepAgent | StepGate | StepAction

export interface WorkflowTrigger {
  type: "manual" | "cron" | "event"
  schedule?: string   // cron expr, only when type === "cron"
  on?: string         // e.g. "task.failed", only when type === "event"
}

export interface WorkflowSpec {
  name: string
  description?: string
  trigger: WorkflowTrigger
  steps: WorkflowStep[]
}

export interface WorkflowRecord {
  slug: string
  filepath: string          // absolute path
  spec: WorkflowSpec
  yaml: string              // raw YAML (for editor display)
  mtime: number             // ms since epoch
}

export interface ValidationError { path: string; message: string }

export function slugifyName(name: string): string
export function parseWorkflow(yaml: string): { spec: WorkflowSpec | null; errors: ValidationError[] }
export function validateWorkflow(spec: unknown): ValidationError[]   // pure, no throw
export function serializeWorkflow(spec: WorkflowSpec): string       // uses yaml.stringify
export function listWorkflows(): Promise<WorkflowRecord[]>          // scans CAE_ROOT/.cae/workflows/*.yml
export function getWorkflow(slug: string): Promise<WorkflowRecord | null>
export function writeWorkflow(spec: WorkflowSpec, opts?: { slug?: string }): Promise<WorkflowRecord>   // mkdir -p + writeFile; slug defaults to slugifyName(spec.name); collision → append -{8hex}
export function WORKFLOWS_DIR(): string                              // returns join(CAE_ROOT, ".cae", "workflows")
```

    1. `import type { WorkflowSpec, WorkflowStep, WorkflowTrigger } from "./cae-workflows"` and `import { validateWorkflow } from "./cae-workflows"` (export `heuristicDraft` as named export).
    2. Structure as ordered rules applied in sequence:
       - **Name derivation:** first sentence (split on `.!?`). Take first 5 words, slugify (inline — same rules as cae-workflows.slugifyName but implemented here to avoid double-import; OR just import slugifyName from cae-workflows). Empty → "new-recipe".

         - `/every monday|every mon/` → schedule base `"0 9 * * 1"`, day 1
         - `/every tuesday|every tue/` → day 2; `/every wednesday|every wed/` → 3; thursday=4, friday=5, saturday=6, sunday=0
         - `/every day|daily/` → `"0 9 * * *"`

         - `/abort|cancel/` alone → push `{ action:"abort" }`
         - `/branch/` → `{ action:"branch" }`
         - Duplicate consecutive steps allowed — the heuristic doesn't deduplicate; output reflects stated intent.

    5. Return valid WorkflowSpec (assert internally for defense: `if (validateWorkflow(result).length > 0) { /* fallback to minimal stub with name preserved */ }`).
