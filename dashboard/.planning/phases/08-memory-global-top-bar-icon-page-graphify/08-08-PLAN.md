    **A) Create `dashboard/scripts/verify-memory-hook.sh`:**

    ```bash
    #!/usr/bin/env bash
    # Phase 8 verification — end-to-end memory-consult hook smoke test.
    #
    # 1. Create a throwaway project dir with an AGENTS.md + .cae/metrics/.
    # 2. Write a task file that asks claude to Read AGENTS.md.
    # 3. Invoke adapters/claude-code.sh with the task.
    # 4. Assert .cae/metrics/memory-consult.jsonl gained >=1 line with the right task_id.
    #
    # Exit 0 on pass; exit 1 on fail. Prints human-readable diagnostics.
    set -uo pipefail

    ADAPTER="/home/cae/ctrl-alt-elite/adapters/claude-code.sh"
    [[ -x "$ADAPTER" ]] || { echo "FAIL: adapter not executable at $ADAPTER"; exit 1; }

    command -v claude >/dev/null 2>&1 || {
      echo "SKIP: claude CLI not on PATH — end-to-end hook test can't run in this environment."
      echo "      Re-run this script on a machine with Claude Code installed."
      exit 0  # SKIP (not FAIL) — keeps the verification doc honest.
    }

    WORK=$(mktemp -d -t phase8-verify-XXXXXX)
    trap 'rm -rf "$WORK"' EXIT

    mkdir -p "$WORK/.cae/metrics"
    cat > "$WORK/AGENTS.md" <<'EOF'
    # AGENTS.md (phase-8-verify fixture)

    - Pattern A: be explicit.
    - Pattern B: avoid implicit state.
    EOF

    TASK_FILE="$WORK/p8-verify.txt"
    cat > "$TASK_FILE" <<'EOF'
    Please read the file AGENTS.md in the current working directory and summarize the two patterns in one short sentence each. Do not do anything else.
    EOF

    echo "> Invoking adapter..."
    pushd "$WORK" >/dev/null
    SESSION_ID="verify-$(date +%s)-$$"
    "$ADAPTER" "$TASK_FILE" "claude-sonnet-4.6" "$SESSION_ID" --system-prompt-file "$TASK_FILE" --timeout 120
    RC=$?
    popd >/dev/null

    echo "> Adapter exit code: $RC"
    [[ $RC -eq 0 ]] || { echo "FAIL: adapter returned non-zero"; exit 1; }

    JSONL="$WORK/.cae/metrics/memory-consult.jsonl"
    if [[ ! -f "$JSONL" ]]; then
      echo "FAIL: $JSONL was not created — hook did not run, or path mismatch."
      echo "  Check: settings.json hook registration, hook script permissions, CAE_TASK_ID export."
      exit 1
    fi

    LINES=$(wc -l < "$JSONL" | tr -d ' ')
    if [[ "$LINES" -lt 1 ]]; then
      echo "FAIL: $JSONL exists but is empty."
      exit 1
    fi

    EXPECTED_TASK_ID="p8-verify"   # basename minus .txt
    if ! grep -q "\"task_id\": \"$EXPECTED_TASK_ID\"" "$JSONL"; then
      echo "FAIL: no line has task_id=\"$EXPECTED_TASK_ID\" — CAE_TASK_ID export may be broken."
      echo "Captured:"
      cat "$JSONL"
      exit 1
    fi

    if ! grep -q '"event": "memory_consult"' "$JSONL"; then
      echo "FAIL: no memory_consult events in $JSONL"
      exit 1
    fi

    if ! grep -q "AGENTS.md" "$JSONL"; then
      echo "FAIL: no AGENTS.md reference in $JSONL — allowlist mismatch?"
      exit 1
    fi

    echo "PASS: $LINES line(s) captured, task_id=$EXPECTED_TASK_ID, AGENTS.md hit."
    exit 0
    ```
