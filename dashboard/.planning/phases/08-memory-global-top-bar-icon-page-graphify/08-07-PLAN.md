```typescript

<Tabs.Root value={view} onValueChange={(v) => setView(v)}>
  <Tabs.List>
    <Tabs.Tab value="browse" className={...}>...</Tabs.Tab>
    <Tabs.Tab value="graph" className={...}>...</Tabs.Tab>
  </Tabs.List>
  <Tabs.Panel value="browse">...</Tabs.Panel>
  <Tabs.Panel value="graph">...</Tabs.Panel>
</Tabs.Root>
```

NO `asChild` (AGENTS.md gotcha). Use `className` only.

Shared state contracts (what MemoryClient owns):
- `view: "browse" | "graph"` — default from URL `?view=`; fallback "browse".
- `selectedPath: string | null` — absolute path of currently-focused memory file (used by BrowsePane selection, NodeDrawer's open-timeline, GitTimelineDrawer's target).
- `activeTaskId: string | null` — taskId for WhyDrawer. Opened via a nav event or prop callback.
- `whyOpen: boolean`, `gitTimelineOpen: boolean`, `gitTimelinePath: string | null`.

Deep-link URL contract:
- `/memory` → view=browse, no selection.
- `/memory?view=graph` → view=graph.
- `/memory?view=browse&path=<abs or rel>` → preselects file in Browse tab.
- `/memory?task=<task_id>` → opens WhyDrawer immediately with that task.

Route params are READ at mount and on `searchParams` change; client component does NOT write back to URL (v1 — URL is one-way-in). Wave 7 may add pushState if the user flow requires it.

    **B) Create `app/memory/memory-client.tsx` (client component):**

    ```typescript
    "use client";
    import { useEffect, useState } from "react";
    import { useSearchParams } from "next/navigation";
    import { Tabs } from "@base-ui/react/tabs";
    import { useDevMode } from "@/lib/providers/dev-mode";
    import { labelFor } from "@/lib/copy/labels";
    import { BrowsePane } from "@/components/memory/browse/browse-pane";
    import { GraphPane } from "@/components/memory/graph/graph-pane";
    import { WhyDrawer } from "@/components/memory/why-drawer";
    import { GitTimelineDrawer } from "@/components/memory/git-timeline-drawer";
    import { ExplainTooltip } from "@/components/ui/explain-tooltip";

    /**
     * Phase 8 — MemoryClient (MEM-01, D-16).
     *
     * Hosts Tabs + both drawers as ONE shared state island. Tabs read their
     * initial value from ?view; drawers open in response to shared state.
     *
     * Drawers must mount here (not in the pane components) because either pane
     * can trigger either drawer — mounting them inside would duplicate state
     * and prevent cross-tab usage.
     */
    type View = "browse" | "graph";

    export default function MemoryClient() {
      const { dev } = useDevMode();
      const L = labelFor(dev);
      const sp = useSearchParams();

      const initialView = (sp.get("view") === "graph" ? "graph" : "browse") as View;
      const initialPath = sp.get("path");
      const initialTask = sp.get("task");

      const [view, setView] = useState<View>(initialView);
      const [selectedPath, setSelectedPath] = useState<string | null>(initialPath);

      // Why drawer
      const [whyOpen, setWhyOpen] = useState<boolean>(Boolean(initialTask));
      const [whyTaskId, setWhyTaskId] = useState<string | null>(initialTask);
      const [whyFilesModified, setWhyFilesModified] = useState<string[] | undefined>(undefined);

      // Git timeline drawer
      const [gitOpen, setGitOpen] = useState<boolean>(false);
      const [gitPath, setGitPath] = useState<string | null>(null);

      // Respond to URL changes after mount (e.g. from TopNav Memory icon with a query)
      useEffect(() => {
        const nextView = (sp.get("view") === "graph" ? "graph" : "browse") as View;
        if (nextView !== view) setView(nextView);
        const nextPath = sp.get("path");
        if (nextPath !== selectedPath) setSelectedPath(nextPath);
        const nextTask = sp.get("task");
        if (nextTask && nextTask !== whyTaskId) {
          setWhyTaskId(nextTask);
          setWhyOpen(true);
        }
        // we deliberately only depend on sp.toString()
        // eslint-disable-next-line react-hooks/exhaustive-deps
      }, [sp.toString()]);

      const openWhy = (taskId: string, filesModified?: string[]) => {
        setWhyTaskId(taskId);
        setWhyFilesModified(filesModified);
        setWhyOpen(true);
      };
      const openGitTimeline = (absPath: string) => {
        setGitPath(absPath);
        setGitOpen(true);
      };

      return (
        <div className="flex flex-1 flex-col gap-3 overflow-hidden">
          <Tabs.Root value={view} onValueChange={(v) => setView(v as View)}>
            <Tabs.List className="inline-flex items-center gap-1 rounded-md border border-[color:var(--border)] bg-[color:var(--surface)] p-1 text-sm">
              <Tabs.Tab
                value="browse"
                data-testid="memory-tab-browse"
                className="rounded px-3 py-1 text-[color:var(--text-muted)] data-[selected]:bg-[color:var(--accent-muted)] data-[selected]:text-[color:var(--accent)]"
              >
                {L.memoryTabBrowse}
              </Tabs.Tab>
              <Tabs.Tab
                value="graph"
                data-testid="memory-tab-graph"
                className="rounded px-3 py-1 text-[color:var(--text-muted)] data-[selected]:bg-[color:var(--accent-muted)] data-[selected]:text-[color:var(--accent)]"
              >
                {L.memoryTabGraph}
              </Tabs.Tab>
              <ExplainTooltip text={L.memoryExplainGraph} ariaLabel="Explain memory tabs" />
            </Tabs.List>
            <Tabs.Panel value="browse" className="flex-1 overflow-hidden">
              <BrowsePane
                initialPath={initialPath ?? undefined}
                selectedPath={selectedPath}
                onSelect={(abs) => setSelectedPath(abs)}
                onOpenGitTimeline={openGitTimeline}
              />
            </Tabs.Panel>
            <Tabs.Panel value="graph" className="flex-1 overflow-hidden">
              <GraphPane onOpenGitTimeline={openGitTimeline} />
            </Tabs.Panel>
          </Tabs.Root>

          <WhyDrawer
            open={whyOpen}
            onClose={() => setWhyOpen(false)}
            taskId={whyTaskId}
            filesModified={whyFilesModified}
          />
          <GitTimelineDrawer
            open={gitOpen}
            onClose={() => setGitOpen(false)}
            absFilePath={gitPath}
          />
        </div>
      );
    }
    ```
