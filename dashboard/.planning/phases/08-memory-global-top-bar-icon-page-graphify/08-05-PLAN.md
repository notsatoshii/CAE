    ```typescript
    import dagre from "@dagrejs/dagre";
    import type { GraphNode, GraphLink } from "@/lib/cae-graph-state";

    const NODE_WIDTH = 180;
    const NODE_HEIGHT = 50;

    /**
     * Phase 8 Wave 3 (D-08, MEM-05): pure dagre layout for the memory graph.
     *
     * Takes graph nodes/edges, runs dagre's LR rank algorithm, returns nodes
     * enriched with `position: { x, y }` suitable for @xyflow/react. The
     * returned array is a new reference — input is NOT mutated.
     *
     * Re-invoked whenever filtered-node set changes (usually cheap; dagre is
     * sync). If perf becomes an issue at the 500-node cap (D-05), memoize in
     * the caller on (filtered-node-id-signature, edge-signature) hashes.
     */
    export function applyDagreLayout(
      nodes: GraphNode[],
      edges: GraphLink[]
    ): Array<GraphNode & { position: { x: number; y: number } }> {
      if (nodes.length === 0) return [];
      const g = new dagre.graphlib.Graph().setDefaultEdgeLabel(() => ({}));
      g.setGraph({ rankdir: "LR", ranksep: 80, nodesep: 40 });
      nodes.forEach((n) => g.setNode(n.id, { width: NODE_WIDTH, height: NODE_HEIGHT }));
      edges.forEach((e) => {
        if (g.hasNode(e.source) && g.hasNode(e.target)) g.setEdge(e.source, e.target);
      });
      dagre.layout(g);
      return nodes.map((n) => {
        const pos = g.node(n.id);
        return { ...n, position: { x: (pos?.x ?? 0) - NODE_WIDTH / 2, y: (pos?.y ?? 0) - NODE_HEIGHT / 2 } };
      });
    }
    ```
