    import { execFile } from "node:child_process";
    import { promisify } from "node:util";
    const execFileP = promisify(execFile);

    // Hard caps (tune carefully — these are security guarantees)
    const MAX_QUERY_LEN = 200;
    const TIMEOUT_MS = 5_000;
    const MAX_BUFFER = 10 * 1024 * 1024;
    const RG_ARGS = ["--json", "--max-count=20", "--max-columns=200", "--glob=*.md", "--smart-case", "--"];

    export async function searchMemory(q: string, rootsOverride?: string[]): Promise<SearchHit[]> {
      if (typeof q !== "string" || q.length === 0) return [];
      if (q.length > MAX_QUERY_LEN) throw new Error("query too long");

      const allowed = await getAllowedRoots(); // from cae-memory-sources
      const roots = (rootsOverride ?? allowed).filter((r) => allowed.includes(r));
      if (roots.length === 0) return [];

      try {
        const { stdout } = await execFileP("rg", [...RG_ARGS, q, ...roots], {
          timeout: TIMEOUT_MS,
          maxBuffer: MAX_BUFFER,
        });
        return parseRgJsonl(stdout);
      } catch (err) {
        const code = (err as { code?: number } | null)?.code;
        if (code === 1) return []; // no match — NOT an error
        throw err;
      }
    }

    function parseRgJsonl(stdout: string): SearchHit[] {
      // ... parse rg's JSONL output: filter type=="match" lines, emit { file, line, preview }
    }
