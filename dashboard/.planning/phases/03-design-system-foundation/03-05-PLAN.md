  breakerHaltedNo: string;                       // "all good" | "running"
  // Metrics tabs
  metricTabBreakers: string;                     // "Pauses" | "Breakers"
  metricTabSentinel: string;                     // "Checks" | "Sentinel"
  metricTabCompaction: string;                   // "Memory cleanup" | "Compaction"
  metricTabApprovals: string;                    // "Approvals" (same in both)
  // Queue
  queueInboxHeading: string;                     // "Waiting to run" | "Inbox"
  queueInboxSub: string;                         // "jobs lined up for CAE" | "awaiting execution"
  queueOutboxHeading: string;                    // "Shipped" | "Outbox"
  queueOutboxSub: string;                        // "finished jobs" | "completed"
  queueColTaskId: string;                        // "Job ID" | "Task ID"
  queueColBuildplan: string;                     // "Instructions" | "BUILDPLAN"
  queueColBranch: string;                        // "Version" | "Branch"
  queueColCommits: string;                       // "Changes" | "Commits"
  // Delegation form
  delegateHeading: string;                       // "Send a job to CAE" | "Delegate to CAE"

  delegateSubmit: string;                        // "Send it" | "Delegate to CAE"
  delegateSubmitPending: string;                 // "Sending…" | "Delegating…"

  phaseDetailHeading: (n: number, name: string) => string; // "{name}" | "Phase {n} — {name}"
  phaseDetailBackLabel: string;                  // "← Back" | "← Build"
  phaseDetailBranchLabel: string;                // "Version" | "Branch"

  mergedCommitsHeading: string;                  // "Shipped changes" | "Merged commits"

  viewOutputButton: string;                      // "See what's happening" | "View output"

}
