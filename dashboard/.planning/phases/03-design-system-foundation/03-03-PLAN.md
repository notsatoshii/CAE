```tsx
"use client";

import { usePathname, useRouter } from "next/navigation";
import { cn } from "@/lib/utils";

type Mode = "plan" | "build";

export function ModeToggle() {
  const pathname = usePathname();
  const router = useRouter();

  // Plan mode when path starts with /plan; Build mode when path starts with /build
  // Memory/Metrics routes do not set an active segment (they're mode-neutral).
  const activeMode: Mode | null = pathname.startsWith("/plan")
    ? "plan"
    : pathname.startsWith("/build")
      ? "build"
      : null;

  function navigate(mode: Mode) {
    document.cookie = `cae-mode=${mode}; max-age=${60 * 60 * 24 * 365}; path=/`;
    router.push(`/${mode}`);
  }

  return (
    <div
      role="tablist"
      aria-label="Mode"
      className="inline-flex items-center gap-0 rounded-md border border-[color:var(--border-subtle)] bg-[color:var(--surface)] p-0.5 text-xs"
    >
      <SegmentButton label="Plan" active={activeMode === "plan"} onClick={() => navigate("plan")} />
      <SegmentButton label="Build" active={activeMode === "build"} onClick={() => navigate("build")} />
    </div>
  );
}

function SegmentButton({ label, active, onClick }: { label: string; active: boolean; onClick: () => void }) {
  return (
    <button
      type="button"
      role="tab"
      aria-selected={active}
      onClick={onClick}
      className={cn(
        "h-6 rounded-[6px] px-2.5 font-medium transition-colors",
        active
          ? "bg-[color:var(--accent-muted)] text-[color:var(--accent)]"
          : "text-[color:var(--text-muted)] hover:text-[color:var(--text)]"
      )}
    >
      {label}
    </button>
  );
}
```

```tsx
"use client";

import { useEffect, useState } from "react";

interface BreakersState {
  inputTokensToday: number;
  outputTokensToday: number;
  halted: boolean;
  retryCount: number;
  activeForgeCount: number;
  recentPhantomEscalations: number;
}

// Rough Claude Sonnet 4.5 estimate (USD / 1M tokens): $3 input, $15 output.
// This IS an estimate per UI-SPEC §S4.6 — labeled "est." in the UI.
function estimateUsd(input: number, output: number): number {
  return (input * 3 + output * 15) / 1_000_000;
}

function formatTokens(n: number): string {
  if (n < 1000) return String(n);
  if (n < 1_000_000) return `${(n / 1000).toFixed(1)}k`;
  return `${(n / 1_000_000).toFixed(2)}M`;
}

export function CostTicker({ projectPath = "" }: { projectPath?: string }) {
  const [stats, setStats] = useState<BreakersState | null>(null);

  useEffect(() => {
    const url = `/api/state?project=${encodeURIComponent(projectPath)}`;
    async function poll() {
      try {
        const res = await fetch(url);
        if (res.ok) {
          const data = await res.json();
          setStats(data.breakers as BreakersState);
        }
      } catch {
        // network blip — keep previous value
      }
    }
    poll();
    const id = setInterval(poll, 3000);
    return () => clearInterval(id);
  }, [projectPath]);

  if (!stats) {
    return (
      <span
        className="inline-flex items-center gap-1.5 font-mono text-xs text-[color:var(--text-muted)]"
        title="Estimates based on local token logs, not Anthropic billing."
      >
        <span>$— · —tok</span>
        <span className="uppercase text-[10px] tracking-wider opacity-70">est.</span>
      </span>
    );
  }

  const totalTokens = stats.inputTokensToday + stats.outputTokensToday;
  const usd = estimateUsd(stats.inputTokensToday, stats.outputTokensToday);

  return (
    <span
      className="inline-flex items-center gap-1.5 font-mono text-xs text-[color:var(--text)]"
      title="Estimates based on local token logs, not Anthropic billing."
    >
      <span>${usd.toFixed(2)}</span>
      <span className="text-[color:var(--text-muted)]">·</span>
      <span>{formatTokens(totalTokens)}tok today</span>
      <span className="uppercase text-[10px] tracking-wider text-[color:var(--text-muted)]">est.</span>
    </span>
  );
}
```

```tsx
"use client";

import { useEffect, useState } from "react";
import { cn } from "@/lib/utils";

interface BreakersState {
  halted: boolean;
  retryCount: number;
  recentPhantomEscalations: number;
}

type Status = "up" | "degraded" | "halt";

export function HeartbeatDot({ projectPath = "" }: { projectPath?: string }) {
  const [status, setStatus] = useState<Status>("up");

  useEffect(() => {
    const url = `/api/state?project=${encodeURIComponent(projectPath)}`;
    async function poll() {
      try {
        const res = await fetch(url);
        if (!res.ok) {
          setStatus("degraded");
          return;
        }
        const data = await res.json();
        const b = data.breakers as BreakersState;
        if (b.halted) setStatus("halt");
        else if (b.retryCount > 0 || b.recentPhantomEscalations > 0) setStatus("degraded");
        else setStatus("up");
      } catch {
        setStatus("degraded");
      }
    }
    poll();
    const id = setInterval(poll, 3000);
    return () => clearInterval(id);
  }, [projectPath]);

  const color = status === "up"
    ? "var(--success)"
    : status === "degraded"
      ? "var(--warning)"
      : "var(--danger)";
  const label = status === "up" ? "Live" : status === "degraded" ? "Degraded" : "Halted";

  return (
    <span className="inline-flex items-center gap-1.5 text-xs text-[color:var(--text-muted)]" title={label}>
      <span
        aria-label={label}
        className={cn(
          "inline-block size-2 rounded-full",
          status === "up" && "animate-pulse"
        )}
        style={{ backgroundColor: color }}
      />
      <span className="font-mono">{status === "up" ? "live" : status}</span>
    </span>
  );
}
```
