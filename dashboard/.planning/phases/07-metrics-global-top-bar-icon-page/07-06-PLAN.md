    **A) Create `dashboard/scripts/lint-no-dollar.sh`:**

    ```bash
    #!/usr/bin/env bash
    # USD-guard: Phase 7 locks metrics copy to tokens-only (D-07).
    # This script exits 1 if any `$` appears in metrics-owned files,
    # excluding template-expression uses (`${...}`).
    #
    # Scope (explicit — expand carefully):
    #   - app/metrics/**
    #   - components/metrics/**
    #   - lib/copy/labels.ts (metrics.* blocks)
    #
    # Out-of-scope (allowed to contain $ — not metrics copy):
    #   - Everything else in the repo.
    set -euo pipefail

    cd "$(dirname "$0")/.."

    FOUND=0

    # Grep for `$` NOT followed by `{` (template expression).
    # -P = Perl regex for negative lookahead.
    # Include only .ts / .tsx / .md files.

    check() {
      local dir_or_file="$1"
      if [[ -e "$dir_or_file" ]]; then
        if grep -rP --include='*.ts' --include='*.tsx' --include='*.md' '(?<!\\)\$(?!\{)' "$dir_or_file" 2>/dev/null; then
          echo "FAIL: literal \$ found in $dir_or_file (D-07: metrics copy is tokens-only)"
          FOUND=1
        fi
      fi
    }

    check app/metrics
    check components/metrics

    # For labels.ts, restrict to metricsXxx lines only (other phases may legitimately use $).
    if [[ -f lib/copy/labels.ts ]]; then
      if grep -nE '^\s*metrics[A-Z][A-Za-z0-9]*' lib/copy/labels.ts | grep -P '(?<!\\)\$(?!\{)' >/dev/null 2>&1; then
        echo "FAIL: literal \$ found in metrics.* labels in lib/copy/labels.ts (D-07)"
        FOUND=1
      fi
    fi

    if [[ $FOUND -eq 0 ]]; then
      echo "lint-no-dollar: PASS (no \$ in metrics copy)"
    fi

    exit $FOUND
    ```
