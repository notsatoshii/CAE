    **A) `dashboard/components/metrics/explain-tooltip.tsx` (NEW):**

    ```tsx
    "use client";

    import { Popover } from "@base-ui/react/popover";
    import { HelpCircle } from "lucide-react";
    import { useExplainMode } from "@/lib/providers/explain-mode-provider";

    interface Props {
      text: string;
      ariaLabel?: string;
    }

    export function ExplainTooltip({ text, ariaLabel }: Props) {
      const { explain } = useExplainMode();

      // When explain-mode is off, render a visually-hidden-but-focusable trigger so keyboard
      // users can still access the explanation. The dev mode toggle doesn't affect this —
      // both devs and founders benefit from per-metric glossary.
      return (
        <Popover.Root>
          <Popover.Trigger
            aria-label={ariaLabel ?? "Explain this metric"}
            className={
              (explain ? "opacity-70 hover:opacity-100 " : "opacity-30 hover:opacity-100 ") +
              "inline-flex h-4 w-4 items-center justify-center rounded-full transition-opacity text-[color:var(--text-muted)] hover:text-[color:var(--accent)] focus-visible:opacity-100 focus-visible:outline-2 focus-visible:outline-[color:var(--accent)]"
            }
            data-testid="explain-trigger"
          >
            <HelpCircle className="size-3.5" aria-hidden="true" />
          </Popover.Trigger>
          <Popover.Portal>
            <Popover.Positioner sideOffset={6}>
              <Popover.Popup
                role="tooltip"
                className="max-w-[260px] rounded-md border border-[color:var(--border)] bg-[color:var(--surface)] p-3 text-xs text-[color:var(--text)] shadow-lg"
                data-testid="explain-popup"
              >
                {text}
              </Popover.Popup>
            </Popover.Positioner>
          </Popover.Portal>
        </Popover.Root>
      );
    }
    ```

    Behavior:
    - Trigger is ALWAYS rendered (not conditionally omitted when explain-mode off) — it dims to 30% opacity but stays focusable. This matches UI-SPEC policy (Explain = default on, but keyboard access never hidden).
    - Popup uses `role="tooltip"` for a11y.
    - `HelpCircle` icon (lucide) at size 3.5 — small, unobtrusive.
