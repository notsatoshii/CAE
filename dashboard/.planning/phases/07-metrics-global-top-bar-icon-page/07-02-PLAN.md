    **Implementation skeleton:**

    ```typescript
    import { readdir, stat } from "fs/promises"
    import { join } from "path"
    import { listProjects, tailJsonl, listInbox, listOutbox } from "./cae-state"
    import { AGENT_META, agentMetaFor } from "./copy/agent-meta"
    import type { CbEvent } from "./cae-types"
    import type { Project } from "./cae-types"

    const CACHE_TTL_MS = 30_000
    let cached: { at: number; value: MetricsState } | null = null

    export async function getMetricsState(): Promise<MetricsState> {
      const now = Date.now()
      if (cached && (now - cached.at) < CACHE_TTL_MS) return cached.value
      const value = await build()
      cached = { at: now, value }
      return value
    }

    async function build(): Promise<MetricsState> {
      const projects = await listProjects()
      const cbEvents: Array<{ project: Project; event: CbEvent }> = []
      const sentinelEvents: Array<{ project: Project; event: Record<string, unknown> }> = []

      for (const p of projects) {
        try {
          const cb = await tailJsonl(join(p.path, ".cae", "metrics", "circuit-breakers.jsonl"), 10000)
          for (const e of cb) {
            if (typeof e === "object" && e !== null && "event" in e) {
              cbEvents.push({ project: p, event: e as CbEvent })
            }
          }
        } catch (err) {
          console.error("[metrics] cb read failed for " + p.name + ":", err)
        }
        try {
          const sn = await tailJsonl(join(p.path, ".cae", "metrics", "sentinel.jsonl"), 2000)
          for (const e of sn) {
            if (typeof e === "object" && e !== null) {
              sentinelEvents.push({ project: p, event: e as Record<string, unknown> })
            }
          }
        } catch {
          // sentinel.jsonl may not exist — silent ok
        }
      }

      return {
        generated_at: new Date().toISOString(),
        spending: buildSpending(cbEvents),
        reliability: buildReliability(cbEvents, sentinelEvents),
        speed: await buildSpeed(cbEvents),
      }
    }
    ```

    **`buildSpending()` — REQ-7-SPEND detail:**

    1. `today = new Date().toISOString().slice(0, 10)`.
    2. `tokens_today` = sum of `e.input_tokens + e.output_tokens` across all cbEvents where `e.ts.startsWith(today)` AND the field exists (typeof-guards). Applies to ANY event kind — the adapter emits them under `token_usage` but future code may fold them into other events.
    3. `tokens_mtd` = sum across events where `e.ts.slice(0, 7) === today.slice(0, 7)` (same calendar month UTC).
    4. `tokens_projected_monthly`:
       ```ts
       const now = new Date()
       const dim = new Date(now.getUTCFullYear(), now.getUTCMonth() + 1, 0).getUTCDate() // days in this UTC month
       const dayOfMonth = now.getUTCDate() // 1-31, current day
       tokens_projected_monthly = Math.round(tokens_mtd * (dim / Math.max(1, dayOfMonth)))
       ```
    5. `by_agent_30d`: one row per DAY in last 30 days (UTC). Each row = `{ date: "YYYY-MM-DD", forge: N, sentinel: M, ... }` with a numeric field per AgentName in AGENT_META. Zero-fill missing agents. Zero-fill missing dates. Derive agent: `e.agent ?? "forge"` (same fallback pattern as Phase 4/5). Skip events with `ts` older than 30d.
    6. `daily_30d`: one row per DAY `{ date, tokens }` — sum of input+output tokens that day across ALL agents. Zero-fill missing dates.
    7. `top_expensive`: flatten events with `task_id` and `input_tokens+output_tokens > 0`; group by `task_id` (sum tokens); sort descending; take top 10; map to `{ task_id, title, tokens, agent, ts }`. `title` = task_id for v0.1 (title-derivation from BUILDPLAN.md across projects is out-of-scope; future enhancement). `agent` = most-frequent agent on the task's events.

    **`buildReliability()` — REQ-7-WELL detail:**

    1. `per_agent_7d`: for each AgentName in AGENT_META:
       - `sevenDaysAgo = Date.now() - 7*86400000`.
       - Filter to `e.event === "forge_end"` for this agent (derive agent: `e.agent ?? "forge"`) with `Date.parse(e.ts) >= sevenDaysAgo`.
       - `sample_n = count`.
       - `success_rate = (count of success:true) / sample_n` (0 if sample_n === 0).
       - Return ALL 9 agents (even if sample_n === 0) — UI decides whether to render; aggregator does not filter. The "min 5 samples" rule is UI policy, not aggregator policy; document in code comment. BUT also emit a `sample_n` field so the UI can apply the ≥5 gate.
    2. `retry_heatmap`: walk events where `e.event === "forge_end"` AND `e.success === false`, OR `e.event === "limit_exceeded"` AND `e.limit === "max_retries"`, within the last 7 days. Bucket by `dow = date.getUTCDay()` (0..6) × `hour = date.getUTCHours()` (0..23). Return the 168 cells (even zero-count — downstream renderer needs the full grid).
    3. `halt_events`: filter `e.event === "halt"`, sort descending by ts, take last 20. Each entry `{ ts, reason: e.reason ?? "unknown", task_id: e.task_id }`.
    4. `sentinel_rejects_30d`: walk sentinelEvents, group by UTC date (last 30 days). For each day:
       - `rejects = count(events where approve === false OR event matches /verdict_invalid|total_failure/)`
       - `approvals = count(events where approve === true)`
       Zero-fill missing dates.

    **`buildSpeed()` — REQ-7-FAST detail:**

    1. `per_agent_wall`: for each agent, pair `forge_begin` events (by `task_id + attempt`) with their matching `forge_end` events. Wall ms = `Date.parse(end.ts) - Date.parse(begin.ts)`. Skip pairings where delta is negative, zero, or > 10*3600000 (10h — guard against stuck sessions). Compute p50 + p95 via linear interpolation:
       ```ts
       function percentile(sorted: number[], p: number): number {
         if (sorted.length === 0) return 0
         const idx = (p / 100) * (sorted.length - 1)
         const lo = Math.floor(idx), hi = Math.ceil(idx)
         if (lo === hi) return sorted[lo]
         return sorted[lo] + (sorted[hi] - sorted[lo]) * (idx - lo)
       }
       ```
       `n` = paired-sample count.
    2. `queue_depth_now`: `(await listInbox()).length`. Single snapshot. Client-side accumulation into a rolling history line is OUT of scope here (UI polish — panel renders current value + sparkline of last 20 polls fed by the client, not the aggregator).
    3. `time_to_merge_bins`: for each DONE outbox task, compute merge time = `outbox DONE.md mtime - inbox task dir creation time` (inbox is ephemeral; if the task was removed from inbox we can't compute — fall back to skipping). Bucket into fixed bins:
       - `"<1m"` — 0..60000 ms
       - `"1-5m"` — 60001..300000
       - `"5-15m"` — 300001..900000
       - `"15m-1h"` — 900001..3600000
       - `">1h"` — else
       Return bins in this fixed order (even if count === 0).

    **Defensive coercion (REQUIRED — same as Phase 5 pattern):**
    - Every event-field read uses `typeof === "number"` / `typeof === "string"` / `typeof === "boolean"` guards before use.
    - Never assume `input_tokens` / `output_tokens` / `success` / `ts` / `task_id` exist on a given event.
    - Wrap per-project reads in try/catch (log to `console.error`, continue).

    **Do NOT:**
    - Read `scribe.jsonl`, `compaction.jsonl`, `approvals.jsonl` — Phase 7 data requirements are fully covered by cb + sentinel.

    - Sort arrays in place on the cached value (mutate a returned cached object = bug) — sort in copies.

    ```typescript
    "use client";

    import { createContext, useContext, useEffect, useRef, useState } from "react";
    import type { MetricsState } from "@/lib/cae-metrics-state";

    interface MetricsPollValue {
      data: MetricsState | null;
      error: Error | null;
    }

    const MetricsPollContext = createContext<MetricsPollValue | null>(null);

    interface MetricsPollProviderProps {
      children: React.ReactNode;
      intervalMs?: number;
    }

    export function MetricsPollProvider({
      children,
      intervalMs = 30_000,
    }: MetricsPollProviderProps) {
      const [data, setData] = useState<MetricsState | null>(null);
      const [error, setError] = useState<Error | null>(null);
      const mounted = useRef(true);

      useEffect(() => {
        mounted.current = true;

        async function poll() {
          // Pause when tab hidden — reduces idle CPU + server load.
          if (typeof document !== "undefined" && document.visibilityState === "hidden") {
            return;
          }
          try {
            const res = await fetch("/api/metrics");
            if (!mounted.current) return;
            if (!res.ok) {
              setError(new Error("/api/metrics returned " + res.status));
              // Still parse the fallback body — route.ts returns full shape on 500.
              try {
                const json = (await res.json()) as MetricsState;
                setData(json);
              } catch {
                // body unreadable; keep prior data
              }
              return;
            }
            const json = (await res.json()) as MetricsState;
            if (!mounted.current) return;
            setData(json);
            setError(null);
          } catch (err) {
            if (!mounted.current) return;
            setError(err instanceof Error ? err : new Error(String(err)));
          }
        }

        // Immediate + scheduled.
        poll();
        const id = window.setInterval(poll, intervalMs);

        // Resume immediately when tab becomes visible again.
        function onVisibility() {
          if (document.visibilityState === "visible") poll();
        }
        document.addEventListener("visibilitychange", onVisibility);

        return () => {
          mounted.current = false;
          window.clearInterval(id);
          document.removeEventListener("visibilitychange", onVisibility);
        };
      }, [intervalMs]);

      return (
        <MetricsPollContext.Provider value={{ data, error }}>
          {children}
        </MetricsPollContext.Provider>
      );
    }

    export function useMetricsPoll(): MetricsPollValue {
      const value = useContext(MetricsPollContext);
      if (value === null) {
        throw new Error("useMetricsPoll must be used inside <MetricsPollProvider>");
      }
      return value;
    }
    ```
