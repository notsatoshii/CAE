Stream-json line schema from `claude --print --output-format stream-json --include-partial-messages` (condensed; see https://docs.claude.com/en/docs/claude-code/sdk):
- Each line is a JSON object; relevant `type` values: `system` (init), `assistant` (full message), `stream_event` with nested `event.type` in {`content_block_start`, `content_block_delta`, `content_block_stop`, `message_delta`, `message_stop`}, and `result` (final).
- Deltas arrive as `{"type":"stream_event","event":{"type":"content_block_delta","delta":{"type":"text_delta","text":"..."}}}`.
- Final usage arrives as `{"type":"result","usage":{"input_tokens":N,"output_tokens":N}}`.
- The spawn wrapper emits raw stdout lines; the route parses and re-frames as SSE.

    ```ts
    import { promises as fs } from "fs";
    import { existsSync } from "fs";
    import { join, resolve } from "path";
    import { randomUUID } from "crypto";
    import { CAE_ROOT } from "./cae-config";
    import type { AgentName } from "./copy/agent-meta";

    export class ValidationError extends Error {
      constructor(msg: string) { super(msg); this.name = "ValidationError"; }
    }

    const UUID_RE = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;

    export function validateSessionId(input: unknown): asserts input is string {
      if (typeof input !== "string") throw new ValidationError("sessionId not a string");
      if (!UUID_RE.test(input)) throw new ValidationError("sessionId not a uuid");
    }

    export function resolveChatPath(sessionId: string): string {
      validateSessionId(sessionId);
      const base = resolve(CAE_ROOT, ".cae", "chat");
      const path = resolve(base, `${sessionId}.jsonl`);
      if (!path.startsWith(base + "/")) throw new ValidationError("path traversal");
      return path;
    }

    async function ensureChatDir(): Promise<void> {
      const dir = resolve(CAE_ROOT, ".cae", "chat");
      if (!existsSync(dir)) await fs.mkdir(dir, { recursive: true });
    }

    export async function getOrCreateSession(agent: AgentName): Promise<string> {
      await ensureChatDir();
      const id = randomUUID();
      const meta: SessionMeta = { session_id: id, agent, created_at: new Date().toISOString() };
      const path = resolveChatPath(id);
      await fs.writeFile(path, JSON.stringify({ role: "meta", ...meta }) + "\n", { flag: "wx" });
      return id;
    }

    export async function appendMessage(sessionId: string, msg: ChatMessage): Promise<void> {
      const path = resolveChatPath(sessionId);
      // atomic line append: fs.appendFile is atomic for lines <4KB under POSIX (gotcha #15)
      await fs.appendFile(path, JSON.stringify(msg) + "\n");
    }

    export async function readTranscript(sessionId: string, limit?: number): Promise<ChatMessage[]> {
      const path = resolveChatPath(sessionId);
      let text: string;
      try { text = await fs.readFile(path, "utf8"); } catch { return []; }
      const lines = text.split("\n").filter(Boolean);
      const msgs: ChatMessage[] = [];
      for (const line of lines) {
        try {
          const obj = JSON.parse(line);
          if (obj.role === "meta") continue;
          if (obj.role === "user" || obj.role === "assistant") msgs.push(obj as ChatMessage);
        } catch { /* skip torn or invalid lines */ }
      }
      if (typeof limit === "number") return msgs.slice(-limit);
      return msgs;
    }

    export async function readTranscriptAfter(sessionId: string, afterMsgId: string | null): Promise<ChatMessage[]> {
      const all = await readTranscript(sessionId);
      if (!afterMsgId) return all;
      const idx = all.findIndex((m) => m.id === afterMsgId);
      if (idx < 0) return [];  // replay-from-unknown = empty (client's last_seen is stale)
      return all.slice(idx + 1);
    }

    export async function getSessionMeta(sessionId: string): Promise<SessionMeta | null> {
      const path = resolveChatPath(sessionId);
      let text: string;
      try { text = await fs.readFile(path, "utf8"); } catch { return null; }
      const first = text.split("\n", 1)[0];
      if (!first) return null;
      try {
        const obj = JSON.parse(first);
        if (obj.role !== "meta") return null;
        return { session_id: obj.session_id, agent: obj.agent, created_at: obj.created_at };
      } catch { return null; }
    }

    export async function setSessionMeta(sessionId: string, meta: SessionMeta): Promise<void> {
      const path = resolveChatPath(sessionId);
      const text = await fs.readFile(path, "utf8");
      const lines = text.split("\n");
      lines[0] = JSON.stringify({ role: "meta", ...meta });
      const tmp = path + ".tmp";
      await fs.writeFile(tmp, lines.join("\n"));
      await fs.rename(tmp, path);
    }

    export async function listSessions(): Promise<SessionSummary[]> {
      await ensureChatDir();
      const dir = resolve(CAE_ROOT, ".cae", "chat");
      const entries = await fs.readdir(dir);
      const out: SessionSummary[] = [];
      for (const f of entries) {
        if (!f.endsWith(".jsonl")) continue;
        const id = f.replace(".jsonl", "");
        if (!UUID_RE.test(id)) continue;
        const p = join(dir, f);
        const st = await fs.stat(p);
        const meta = await getSessionMeta(id);
        if (!meta) continue;
        const all = await readTranscript(id);
        const last = all[all.length - 1];
        const preview = last ? last.content.slice(0, 80) : "";
        out.push({
          session_id: id,
          agent: meta.agent,
          created_at: meta.created_at,
          mtime_ms: st.mtimeMs,
          last_preview: preview,
          message_count: all.length,
        });
      }
      out.sort((a, b) => b.mtime_ms - a.mtime_ms);
      return out;
    }
    ```

    ```ts
    import { spawn } from "child_process";
    import type { Readable } from "stream";

    export interface SpawnChatInput {
      sessionId: string;
      voiceFile: string;
      model: string;
      messageText: string;
      cwd: string;
    }

    export interface SpawnChatHandle {
      stdout: Readable;
      stderr: Readable;
      wait: () => Promise<number>;
      kill: () => void;
    }

    /**
     * Spawns `claude --print --resume <sessionId> --append-system-prompt-file <voiceFile>
     *                --output-format stream-json --include-partial-messages --model <model>`
     * with messageText piped to stdin. cwd MUST be CAE_ROOT so the session file lives at the
     * expected `~/.claude/projects/-home-cae-ctrl-alt-elite/<uuid>.jsonl` path (gotcha #4).
     *
     * The returned stdout is a line-delimited JSON stream; callers parse line-by-line.
     * stderr is exposed so the API route can detect rate-limit strings (gotcha #12).
     */
    export function spawnClaudeChat(input: SpawnChatInput): SpawnChatHandle {
      const args = [
        "--print",
        "--resume", input.sessionId,
        "--append-system-prompt-file", input.voiceFile,
        "--output-format", "stream-json",
        "--include-partial-messages",
        "--model", input.model,
      ];
      const child = spawn("claude", args, {
        cwd: input.cwd,
        stdio: ["pipe", "pipe", "pipe"],
        env: { ...process.env },
      });
      // Feed the message over stdin, then close.
      child.stdin.write(input.messageText);
      child.stdin.end();

      let exited: number | null = null;
      const waiters: Array<(code: number) => void> = [];
      child.on("exit", (code) => {
        exited = code ?? 0;
        for (const w of waiters) w(exited);
        waiters.length = 0;
      });

      return {
        stdout: child.stdout!,
        stderr: child.stderr!,
        wait() {
          return new Promise<number>((resolve) => {
            if (exited !== null) resolve(exited);
            else waiters.push(resolve);
          });
        },
        kill() { try { child.kill("SIGTERM"); } catch { /* ignore */ } },
      };
    }
    ```

    ```ts
    export const dynamic = "force-dynamic";

    import { NextRequest } from "next/server";
    import { randomUUID } from "crypto";
    import { join } from "path";
    import { auth } from "@/auth";
    import { CAE_ROOT } from "@/lib/cae-config";
    import { validateSessionId, appendMessage, getSessionMeta, setSessionMeta } from "@/lib/cae-chat-state";
    import { pickPersona, modelForAgent } from "@/lib/voice-router";
    import { spawnClaudeChat } from "@/lib/chat-spawn";
    import type { AgentName } from "@/lib/copy/agent-meta";

    const MAX_MESSAGE_LEN = 4000;

    function encodeSSE(id: string, event: string, data: unknown): string {
      return `id: ${id}\nevent: ${event}\ndata: ${JSON.stringify(data)}\n\n`;
    }

    export async function POST(req: NextRequest) {
      const session = await auth();
      if (!session) return new Response("unauthorized", { status: 401 });

      let body: { sessionId?: string; message?: string; route?: string; on_route?: string };
      try { body = await req.json(); } catch { return Response.json({ error: "bad json" }, { status: 400 }); }
      const { sessionId, message, route } = body;
      const onRoute = body.on_route ?? route ?? "/";
      if (!sessionId || !message || typeof message !== "string") {
        return Response.json({ error: "missing fields" }, { status: 400 });
      }
      if (message.length > MAX_MESSAGE_LEN) return Response.json({ error: "too long" }, { status: 400 });
      try { validateSessionId(sessionId); } catch { return Response.json({ error: "bad sessionId" }, { status: 400 }); }

      // Resolve persona: session meta default; override iff first token is @agent (gotcha #9)
      const meta = await getSessionMeta(sessionId);
      const overrideAgent: AgentName | null = (() => {
        const trimmed = message.trimStart();
        const m = trimmed.match(/^@(nexus|forge|sentinel|scout|scribe|phantom|aegis|arch|herald)\b/i);
        return m ? (m[1].toLowerCase() as AgentName) : null;
      })();
      const agent: AgentName = overrideAgent ?? meta?.agent ?? pickPersona({ route: route ?? "/", message });
      const model = modelForAgent(agent);
      const voiceFile = join(CAE_ROOT, "dashboard", "docs", "voices", `${agent}.md`);

      // Persist override into meta so the NEXT turn reads the same agent.
      if (overrideAgent && meta && overrideAgent !== meta.agent) {
        await setSessionMeta(sessionId, { ...meta, agent: overrideAgent });
      } else if (!meta) {
        // No meta yet (session created elsewhere without meta, or race). Write one.
        await setSessionMeta(sessionId, { session_id: sessionId, agent, created_at: new Date().toISOString() });
      }

      // Append user message to jsonl with a per-message id.
      const userMsgId = randomUUID();
      await appendMessage(sessionId, {
        id: userMsgId, ts: new Date().toISOString(), role: "user", content: message, route: route ?? "/",
      });

      // Spawn claude-chat.
      const handle = spawnClaudeChat({ sessionId, voiceFile, model, messageText: message, cwd: CAE_ROOT });

      const enc = new TextEncoder();
      const stream = new ReadableStream<Uint8Array>({
        async start(sc) {
          const beginId = randomUUID();
          sc.enqueue(enc.encode(encodeSSE(beginId, "assistant.begin", { sessionId, agent, model })));

          let finalText = "";
          let inTokens = 0;
          let outTokens = 0;
          let rateLimited = false;
          let buffer = "";

          handle.stdout.on("data", (chunk: Buffer) => {
            buffer += chunk.toString("utf8");
            let idx;
            while ((idx = buffer.indexOf("\n")) >= 0) {
              const line = buffer.slice(0, idx);
              buffer = buffer.slice(idx + 1);
              if (!line.trim()) continue;
              try {
                const obj = JSON.parse(line);
                // Stream-json text deltas:
                if (obj?.type === "stream_event" && obj?.event?.type === "content_block_delta") {
                  const text = obj?.event?.delta?.text ?? "";
                  if (text) {
                    finalText += text;
                    const deltaId = randomUUID();
                    sc.enqueue(enc.encode(encodeSSE(deltaId, "assistant.delta", { delta: text })));
                    if (onRoute !== "/chat") {
                      sc.enqueue(enc.encode(encodeSSE(randomUUID(), "unread_tick", { unread: 1 })));
                    }
                  }
                } else if (obj?.type === "result") {
                  inTokens = obj?.usage?.input_tokens ?? 0;
                  outTokens = obj?.usage?.output_tokens ?? 0;
                }
              } catch { /* skip non-JSON lines */ }
            }
          });

          let stderrBuf = "";
          handle.stderr.on("data", (chunk: Buffer) => {
            stderrBuf += chunk.toString("utf8");
            if (/rate.?limit/i.test(stderrBuf) || /usage.?limit/i.test(stderrBuf)) {
              rateLimited = true;
            }
          });

          const code = await handle.wait();
          if (rateLimited || code !== 0) {
            sc.enqueue(enc.encode(encodeSSE(randomUUID(), "rate_limited", { retry_after_sec: 30 })));
            sc.close();
            return;
          }

          // Persist assistant message.
          const assistantMsgId = randomUUID();
          await appendMessage(sessionId, {
            id: assistantMsgId, ts: new Date().toISOString(), role: "assistant",
            content: finalText, agent, tokens: { in: inTokens, out: outTokens },
          });

          sc.enqueue(enc.encode(encodeSSE(assistantMsgId, "assistant.end", {
            final: finalText, tokens: { in: inTokens, out: outTokens },
          })));
          sc.close();
        },
        cancel() { handle.kill(); },
      });

      return new Response(stream, {
        headers: {
          "Content-Type": "text/event-stream",
          "Cache-Control": "no-cache",
          "Connection": "keep-alive",
        },
      });
    }
    ```
