    ```tsx
    "use client";

    import { createContext, useCallback, useContext, useEffect, useRef, useState, type ReactNode } from "react";

    export interface ChatRailContextValue {
      open: boolean;
      unread: number;
      currentSessionId: string | null;
      streaming: boolean;
      lastMessagePreview: string;
      lastSeenMsgId: string | null;
      toggle: () => void;
      expand: () => void;
      collapse: () => void;
      setCurrentSession: (id: string | null) => void;
      setLastSeenMsgId: (id: string) => void;
      setLastMessagePreview: (s: string) => void;
      markAllRead: () => void;
      bumpUnread: (opts?: { autoExpand?: boolean }) => void;
      setStreaming: (v: boolean) => void;
      sessionAuthed: boolean;   // false if provider mounted without session → all ops no-op
    }

    const DEFAULT: ChatRailContextValue = {
      open: false, unread: 0, currentSessionId: null, streaming: false, lastMessagePreview: "",
      lastSeenMsgId: null,
      toggle: () => {}, expand: () => {}, collapse: () => {},
      setCurrentSession: () => {}, setLastSeenMsgId: () => {}, setLastMessagePreview: () => {},
      markAllRead: () => {}, bumpUnread: () => {}, setStreaming: () => {},
      sessionAuthed: false,
    };

    const ChatRailContext = createContext<ChatRailContextValue>(DEFAULT);

    export function useChatRail(): ChatRailContextValue {
      return useContext(ChatRailContext);
    }

    export function ChatRailProvider({
      children,
      session,
    }: {
      children: ReactNode;
      session: unknown | null;   // Session from next-auth; we only care if truthy
    }) {
      // If no session, provide the DEFAULT (no-op) context so /signin + unauth pages still render but rail is invisible.
      if (!session) {
        return <ChatRailContext.Provider value={DEFAULT}>{children}</ChatRailContext.Provider>;
      }

      const [open, setOpen] = useState(false);
      const [unread, setUnread] = useState(0);
      const [currentSessionId, setCurrentSessionId] = useState<string | null>(null);
      const [streaming, setStreaming] = useState(false);
      const [lastMessagePreview, setLastMessagePreview] = useState("");
      const [lastSeenMsgId, setLastSeenMsgIdState] = useState<string | null>(null);
      const lastUserCollapseTs = useRef<number>(0);

      const expand = useCallback(() => { setOpen(true); setUnread(0); }, []);
      const collapse = useCallback(() => { setOpen(false); lastUserCollapseTs.current = Date.now(); }, []);
      const toggle = useCallback(() => {
        setOpen((prev) => { if (prev) lastUserCollapseTs.current = Date.now(); return !prev; });
        setUnread((u) => (open ? u : 0));   // opening clears unread
      }, [open]);

      const markAllRead = useCallback(() => setUnread(0), []);
      const bumpUnread = useCallback((opts?: { autoExpand?: boolean }) => {
        setOpen((prev) => {
          if (prev) return prev;       // already open → no unread bump
          if (opts?.autoExpand && Date.now() - lastUserCollapseTs.current >= 500) {
            return true;                // auto-expand (gotcha #13 debounce)
          }
          return prev;
        });
        setUnread((u) => (open ? u : u + 1));
      }, [open]);

      const setCurrentSession = useCallback((id: string | null) => setCurrentSessionId(id), []);
      const setLastSeenMsgId = useCallback((id: string) => setLastSeenMsgIdState(id), []);

      // Escape key collapses when open:
      useEffect(() => {
        if (!open) return;
        function onKey(e: KeyboardEvent) {
          if (e.key === "Escape") {
            e.preventDefault();
            collapse();
          }
        }
        window.addEventListener("keydown", onKey);
        return () => window.removeEventListener("keydown", onKey);
      }, [open, collapse]);

      const value: ChatRailContextValue = {
        open, unread, currentSessionId, streaming, lastMessagePreview, lastSeenMsgId,
        toggle, expand, collapse, setCurrentSession, setLastSeenMsgId, setLastMessagePreview,
        markAllRead, bumpUnread, setStreaming, sessionAuthed: true,
      };

      return <ChatRailContext.Provider value={value}>{children}</ChatRailContext.Provider>;
    }
    ```

    ```tsx
          // Fetch history:
          const h = await fetch(`/api/chat/history/${id}`).then((r) => r.json());
          if (h.messages) setMessages(h.messages);
          if (h.meta?.agent) setCurrentAgent(h.meta.agent);
        })();
      }, []); // eslint-disable-line react-hooks/exhaustive-deps

      // Auto-scroll on new message:
      useEffect(() => {
        scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight, behavior: "smooth" });
      }, [messages.length]);

      async function send(text: string) {
        if (!text.trim()) return;
        if (sending) return;
        if (Date.now() < rateLimitUntil) return;
        const sessionId = rail.currentSessionId;
        if (!sessionId) return;
        setSending(true);
        rail.setStreaming(true);
        setDraft("");

        const userMsg: ChatMessageUI = { id: crypto.randomUUID(), role: "user", content: text, ts: new Date().toISOString() };
        setMessages((ms) => [...ms, userMsg]);
        const assistantDraft: ChatMessageUI = { id: crypto.randomUUID(), role: "assistant", content: "", agent: currentAgent, ts: new Date().toISOString() };
        setMessages((ms) => [...ms, assistantDraft]);

        const controller = new AbortController();
        abortRef.current = controller;
        try {
          const res = await fetch("/api/chat/send", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ sessionId, message: text, route: pathname, on_route: pathname }),
            signal: controller.signal,
          });
          if (!res.ok || !res.body) throw new Error("send failed");
          const reader = res.body.getReader();
          const decoder = new TextDecoder();
          let buf = "";
          let currentId: string | null = null;
          while (true) {
            const { value, done } = await reader.read();
            if (done) break;
            buf += decoder.decode(value, { stream: true });
            // Parse SSE frames separated by \n\n:
            let idx;
            while ((idx = buf.indexOf("\n\n")) >= 0) {
              const frame = buf.slice(0, idx);
              buf = buf.slice(idx + 2);
              const lines = frame.split("\n");
              let id: string | null = null, event: string | null = null, data: string | null = null;
              for (const l of lines) {
                if (l.startsWith("id: ")) id = l.slice(4);
                else if (l.startsWith("event: ")) event = l.slice(7);
                else if (l.startsWith("data: ")) data = l.slice(6);
              }
              if (!event) continue;
              // D-17 de-dupe: skip if id <= lastSeenMsgId
              if (id && rail.lastSeenMsgId && id === rail.lastSeenMsgId) continue;
              if (id) { currentId = id; rail.setLastSeenMsgId(id); }
              try {
                const obj = data ? JSON.parse(data) : {};
                if (event === "assistant.begin") {
                  if (obj.agent) setCurrentAgent(obj.agent);
                } else if (event === "assistant.delta") {
                  setMessages((ms) => {
                    const last = ms[ms.length - 1];
                    if (!last || last.role !== "assistant") return ms;
                    return [...ms.slice(0, -1), { ...last, content: last.content + (obj.delta ?? "") }];
                  });
                  // If collapsed, bump unread (D-09):
                  rail.bumpUnread({ autoExpand: true });
                } else if (event === "assistant.end") {
                  setMessages((ms) => {
                    const last = ms[ms.length - 1];
                    if (!last || last.role !== "assistant") return ms;
                    return [...ms.slice(0, -1), { ...last, content: obj.final ?? last.content }];
                  });
                } else if (event === "rate_limited") {
                  setRateLimitUntil(Date.now() + (obj.retry_after_sec ?? 30) * 1000);
                } else if (event === "unread_tick") {
                  rail.bumpUnread();
                }
              } catch { /* skip malformed */ }
            }
          }
        } catch {
          /* already handled via rate_limited / network errors surface in UI */
        } finally {
          setSending(false);
          rail.setStreaming(false);
          abortRef.current = null;
        }
      }

      const rateLimitSecs = rateLimitUntil > Date.now() ? Math.ceil((rateLimitUntil - Date.now()) / 1000) : 0;

      return (
        <div className={standalone ? "flex flex-col h-full max-w-[800px] mx-auto" : "flex flex-col h-full"}>
          <div ref={scrollRef} className="flex-1 overflow-y-auto px-3 py-2 space-y-1">
            {messages.length === 0 ? (
              <p className="text-sm text-[color:var(--text-muted,#8a8a8c)]">{t.chatEmptyThread}</p>
            ) : (
              messages.map((m) => <Message key={m.id} role={m.role} content={m.content} agent={m.agent ?? null} ts={m.ts} streaming={sending && m === messages[messages.length - 1] && m.role === "assistant"} />)
            )}
          </div>
          {rateLimitSecs > 0 ? (
            <p className="px-3 py-1 text-xs text-[color:var(--warning,#f59e0b)] border-t border-[color:var(--border,#1f1f22)]">
              {t.chatRateLimited(rateLimitSecs)}
            </p>
          ) : null}
          <Suggestions onPick={(msg) => send(msg)} />
          <form
            onSubmit={(e) => { e.preventDefault(); void send(draft); }}
            className="flex gap-2 px-3 py-2 border-t border-[color:var(--border,#1f1f22)]"
          >
            <textarea
              value={draft}
              onChange={(e) => setDraft(e.target.value)}
              onKeyDown={(e) => {
                if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); void send(draft); }
              }}
              disabled={sending || rateLimitSecs > 0}
              placeholder={t.chatInputPlaceholder}
              rows={2}
              className="flex-1 resize-none text-sm bg-transparent border border-[color:var(--border,#1f1f22)] rounded p-1 text-[color:var(--text,#e5e5e5)]"
            />
            <button type="submit" disabled={sending || !draft.trim() || rateLimitSecs > 0} className="text-sm px-3 py-1 rounded bg-[color:var(--accent,#00d4ff)] text-black disabled:opacity-50">
              {sending ? t.chatSendButtonPending : t.chatSendButton}
            </button>
          </form>
        </div>
      );
    }
    ```
