    ```ts
    export type ChatGatedActionSpec =
      | { type: "workflow_run"; slug: string; priorRuns?: number[] }
      | { type: "delegate_new"; buildplanLength?: number }
      | { type: "retry_task"; priorAttempts?: number[] }
      | { type: "chat_send" };

    export function estimateTokens(action: ChatGatedActionSpec): number {
      switch (action.type) {
        case "chat_send":
          return 0;
        case "delegate_new":
          return 8000;
        case "workflow_run": {
          const runs = action.priorRuns ?? [];
          if (runs.length === 0) return 10000;
          const avg = runs.reduce((a, b) => a + b, 0) / runs.length;
          return Math.round(avg);
        }
        case "retry_task": {
          const atts = action.priorAttempts ?? [];
          if (atts.length === 0) return 5000;
          const avg = atts.reduce((a, b) => a + b, 0) / atts.length;
          return Math.round(avg);
        }
      }
    }

    export function shouldGate(action: ChatGatedActionSpec): boolean {
      if (action.type === "chat_send") return false;
      return estimateTokens(action) >= 1000;
    }
    ```
