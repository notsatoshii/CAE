    **C) `components/changes/day-group.tsx`** — day headers grouped from flat events.

    ```tsx
    "use client";

    import { useDevMode } from "@/lib/providers/dev-mode";
    import { labelFor } from "@/lib/copy/labels";
    import { relativeTime } from "@/lib/cae-changes-state";
    import { ChangeRow } from "./change-row";
    import type { ChangeEvent } from "@/lib/cae-changes-state";

    function dayBucket(iso: string, now: Date = new Date()): "today" | "yesterday" | "week" | "older" {
      const t = relativeTime(iso, now);
      if (["just now", "this morning", "this afternoon", "this evening"].includes(t)) return "today";
      if (t === "yesterday") return "yesterday";
      // 3-letter day name OR M/D → week if day-name, older if M/D
      if (/^[A-Z][a-z]+$/.test(t)) return "week";
      return "older";
    }

    export function DayGroup({ events }: { events: ChangeEvent[] }) {
      const { dev } = useDevMode();
      const t = labelFor(dev);
      // Group by bucket preserving order:
      const today: ChangeEvent[] = [];
      const yesterday: ChangeEvent[] = [];
      const thisWeek = new Map<string, ChangeEvent[]>();   // keyed by day name
      const older = new Map<string, ChangeEvent[]>();      // keyed by M/D
      for (const e of events) {
        const b = dayBucket(e.ts);
        const rt = relativeTime(e.ts);
        if (b === "today") today.push(e);
        else if (b === "yesterday") yesterday.push(e);
        else if (b === "week") {
          if (!thisWeek.has(rt)) thisWeek.set(rt, []);
          thisWeek.get(rt)!.push(e);
        } else {
          if (!older.has(rt)) older.set(rt, []);
          older.get(rt)!.push(e);
        }
      }
      const sections: Array<{ label: string; evs: ChangeEvent[] }> = [];
      if (today.length) sections.push({ label: t.changesDayToday, evs: today });
      if (yesterday.length) sections.push({ label: t.changesDayYesterday, evs: yesterday });
      for (const [day, evs] of thisWeek.entries()) sections.push({ label: t.changesDayWeek(day), evs });
      for (const [day, evs] of older.entries()) sections.push({ label: day, evs });

      return (
        <div className="space-y-4">
          {sections.map(({ label, evs }) => (
            <div key={label}>
              <h3 className="text-sm font-medium text-[color:var(--text-muted,#8a8a8c)] mb-2">{label}</h3>
              <div className="divide-y divide-[color:var(--border,#1f1f22)]">
                {evs.map((e) => <ChangeRow key={e.sha} event={e} />)}
              </div>
            </div>
          ))}
        </div>
      );
    }
    ```

    ```tsx
    "use client";

    import { useEffect, useState } from "react";
    import { Accordion } from "@base-ui-components/react/accordion";
    import { useDevMode } from "@/lib/providers/dev-mode";
    import { labelFor } from "@/lib/copy/labels";
    import { ExplainTooltip } from "@/components/ui/explain-tooltip";
    import { ProjectGroup } from "@/components/changes/project-group";
    import type { ProjectGroup as ProjectGroupData } from "@/lib/cae-changes-state";

    interface ChangesResponse {
      projects: ProjectGroupData[];
      generated_at: string;
      cache_ttl_ms: number;
    }

    export function ChangesClient() {
      const { dev } = useDevMode();
      const t = labelFor(dev);
      const [data, setData] = useState<ChangesResponse | null>(null);
      const [error, setError] = useState<string | null>(null);

      useEffect(() => {
        let cancelled = false;
        (async () => {
          try {
            const res = await fetch("/api/changes");
            if (!res.ok) throw new Error("status " + res.status);
            const j = (await res.json()) as ChangesResponse;
            if (!cancelled) setData(j);
          } catch (err) {
            if (!cancelled) setError(err instanceof Error ? err.message : "fetch failed");
          }
        })();
        return () => { cancelled = true; };
      }, []);

      if (error) return <p className="text-sm text-destructive">{t.changesFailedToLoad}</p>;
      if (!data) return <p className="text-sm text-[color:var(--text-muted,#8a8a8c)]">{t.changesEmpty}</p>;

      const totalToday = data.projects.reduce((acc, g) => {
        return acc + g.events.filter((e) => {
          const d = new Date(e.ts);
          const now = new Date();
    ```
