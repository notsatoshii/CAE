NEW — dashboard/lib/floor/renderer.ts:
```ts
import type { Scene } from "./scene";

export interface Viewport {
  width: number;   // canvas width in CSS px
  height: number;  // canvas height in CSS px
  cx: number;      // camera x — typically width/2
  cy: number;      // camera y — typically height/2 - 80 (research recommendation)
  devLabels: boolean; // true = dev copy, false = founder copy (from useDevMode at call site)
}

/**
 * Pure draw routine. Draws in this order:
 *   1. dark-bg fill (var(--bg) === #0a0a0a, encoded as literal "#0a0a0a" since ctx has no CSS var access)
 *   2. stations — z-sorted by (tx + ty) ascending so back stations render first
 *      each station: diamond rect fill (tinted by status) + text label below
 *   3. effects — rendered in `scene.effects` array order
 *   4. entities — phantoms drawn ON TOP of their current lerp position
 *
 * Never throws; every branch has a default fill. Never reads time — animations
 * are driven by `effect.ttl` / `entity` state which the caller mutates via step().
 *
 * Side-effects: ONLY ctx draw calls. No console, no globals, no performance.now(),
 * no requestAnimationFrame. This is what makes it unit-testable with a stub ctx.
 */
export function render(
  ctx: CanvasRenderingContext2D,
  scene: Scene,
  viewport: Viewport
): void;
```
