"use client";

import { MutableRefObject } from "react";
import type { Scene } from "@/lib/floor/scene";

export interface UseFloorEventsOpts {
  /** Absolute path. null/undefined = no SSE opened (idle). */
  cbPath: string | null;
  /** When true, incoming events are queued but not applied to scene. */
  paused: boolean;
  /** Hook mutates sceneRef.current directly (effects push, status updates). */
  sceneRef: MutableRefObject<Scene>;
}

export interface UseFloorEventsResult {
  /** Count of active effects in scene (updated each React frame via a state mirror). */
  effectsCount: number;
  /** Count of pending events in queue (not yet applied). */
  queueSize: number;
  /** Epoch ms of the most recently-applied event, or null if none yet. */
  lastEventTs: number | null;
  /** True when /api/state returned 401 on the drift probe; toolbar shows re-auth banner. */
  authDrifted: boolean;
}

/**
 * Core wiring for the Live Floor. Wraps /api/tail SSE + parseEvent + mapEvent +
 * queue/effects caps + reduced-motion gate + 30s /api/state auth-drift probe.
 *
 * Returns observable counters (for the toolbar). The hook DOES NOT own the RAF
 * loop — that's the canvas component's job — but it DOES own the queue drain
 * into `sceneRef.current.effects` (so it can enforce EFFECTS_CAP consistently
 * regardless of RAF cadence).

 * Implementation: one useEffect for SSE (deps: [cbPath]), one for auth-drift
 * interval (deps: [cbPath]), one for queue-drain microtask / rAF poke
 * (deps: [paused]). Queue drain uses `queueMicrotask` so caps are enforced
 * even if the component's RAF is slow/paused.
 */
export function useFloorEvents(opts: UseFloorEventsOpts): UseFloorEventsResult;

  MAX_LINE_BYTES: 4096;

<open_questions>
**Q1 — `useFloorEvents` queue drain cadence: microtask vs RAF?**
Decision: **queueMicrotask inside onmessage**. The hook drains synchronously on receipt, applies MappedEffects to scene, enforces EFFECTS_CAP. This decouples event handling from render cadence so caps hold even when the canvas is unmounted briefly (e.g. pop-out detach). The canvas only READS scene; it never mutates scene itself from event data.

**Q2 — Auth-drift probe: `/api/state` vs a lighter `/api/health`?**
Decision: **/api/state**. Already exists, auth-gated identically to /api/tail, no new route needed. 30s interval is well within /api/state's poll cadence (already hit every 3s by use-state-poll elsewhere); adding a second consumer is trivial.

**Q3 — How does the canvas stay in sync with scene after hook refactor?**
Decision: the canvas owns `sceneRef`; it PASSES the ref to the hook. The hook mutates `sceneRef.current`. The canvas's RAF loop reads `sceneRef.current` each tick (unchanged from Plan 02). React counter values (`effectsCount`, etc.) are surfaced via the hook's returned object and forwarded to `props.onMetrics?.()` each React render — these are OBSERVATIONS only; the source of truth remains the mutable scene.
</open_questions>
